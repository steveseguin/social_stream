<script>
(function () {
  // Configuration
  var startSeconds = 5 * 60; // 5 minutes
  var remainingSeconds = startSeconds;
  var timerElement = document.getElementById('countdown');
  var intervalId = null;

  // Helper to format time MM:SS
  function formatTime(totalSeconds) {
    var m = Math.floor(totalSeconds / 60);
    var s = totalSeconds % 60;
    return (m < 10 ? '0' + m : m) + ':' + (s < 10 ? '0' + s : s);
  }

  // Update the display
  function updateDisplay() {
    timerElement.textContent = formatTime(remainingSeconds);
  }

  // Start the countdown
  function startCountdown() {
    updateDisplay();
    intervalId = setInterval(function () {
      remainingSeconds--;
      if (remainingSeconds < 0) {
        clearInterval(intervalId);
        timerElement.textContent = "LIVE";
        timerElement.style.color = "#22c55e"; // Green for live
        document.querySelector('h1').textContent = "We Are Live!";
      } else {
        updateDisplay();
      }
    }, 1000);
  }

  // Initialize
  startCountdown();

  // VDO.Ninja Bridge Setup (Standard SSN Pattern)
  // This ensures the overlay is ready to receive data if a session is connected,
  // though this specific "Starting Soon" page is mostly autonomous.
  (function () {
    var params = new URLSearchParams(location.search);
    var roomID = params.get("session") || "";
    var password = params.get("password") || "false";
    var label = params.get("label") || "starting_soon_overlay_" + Date.now();
    var bridge = document.createElement("iframe");
    bridge.id = "ssn_bridge";
    bridge.style.display = "none";
    document.body.appendChild(bridge);

    if (roomID) {
      bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
        encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
        "&label=" + encodeURIComponent(label) +
        "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
    }

    window.addEventListener("message", function (event) {
      var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
      if (typeof payload !== "undefined") {
        // If a specific event (like "start_now") is received, we could stop the timer here.
        // For now, we just acknowledge it.
        console.log("SSN Data Received:", payload);
      }
    });
  })();
})();
</script>
