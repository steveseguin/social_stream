  .alert-card[data-type="subscription_gift"] { border-color: var(--accent-gift); }
  .alert-card[data-type="gifted"] { border-color: var(--accent-gift); }
  .alert-card[data-type="donation"] { border-color: var(--accent-donate); }

<script>
(function () {
  // --- Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_overlay_" + Date.now();
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // --- State & Config ---
  var container = document.getElementById("alert-container");
  var DISMISS_DELAY = 4000; // 4 seconds

  // --- Helper: Escape HTML ---
  function escapeHtml(str) {
    if (!str) return "";
    return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&#039;");
  }

  // --- Helper: Create Alert Element ---
  function createAlertCard(type, name, detail, avatarUrl) {
    var card = document.createElement("div");
    card.className = "alert-card";
    card.setAttribute("data-type", type);

    // Avatar
    if (avatarUrl) {
      var img = document.createElement("img");
      img.className = "alert-avatar";
      img.src = avatarUrl;
      img.onerror = function() { this.style.display = 'none'; };
      card.appendChild(img);
    }

    // Name
    var nameSpan = document.createElement("span");
    nameSpan.className = "alert-name";
    nameSpan.textContent = name || "Unknown";
    card.appendChild(nameSpan);

    // Detail (Message, Amount, Count, etc.)
    if (detail) {
      var detailSpan = document.createElement("span");
      detailSpan.className = "alert-detail";
      detailSpan.textContent = detail;
      card.appendChild(detailSpan);
    }

    return card;
  }

  // --- Core Handler ---
  function handlePayload(data) {
    if (!data) return;

    var type = data.type || "";
    var event = data.event || "";
    var name = data.chatname || "";
    var message = data.chatmessage || "";
    var donation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";
    var avatar = data.chatimg || "";
    var alertType = null;
    var detailText = "";

    // 1. New Follower
    if (event === "new_follower") {
      alertType = "new_follower";
      detailText = "just followed!";
    }
    // 2. Subscriber (Resub)
    else if (event === "subscriber" || event === "resub") {
      alertType = "subscriber";
      detailText = membership ? membership + " — " + subtitle : "subscribed!";
    }
    // 3. Gifted Sub
    else if (event === "subscription_gift" || event === "gifted") {
      alertType = "subscription_gift";
      // Try to parse count from subtitle or meta
      var count = subtitle;
      if (data.meta && data.meta.gift_count) count = data.meta.gift_count;
      detailText = "gifted " + (count || "a") + " sub(s)!";
    }
    // 4. Donation
    else if (donation) {
      alertType = "donation";
      detailText = "donated " + donation;
      if (message) detailText += " — " + message;
    }
    // 5. Raid
    else if (event === "raid") {
      alertType = "raid";
      var viewers = subtitle;
      if (data.meta && data.meta.viewers) viewers = data.meta.viewers;
      detailText = "is raiding with " + (viewers || "many") + " viewers!";
    }

    // If we identified an alert type, render it
    if (alertType) {
      var card = createAlertCard(alertType, name, detailText, avatar);
      container.appendChild(card);

      // Auto-dismiss
      setTimeout(function () {
        if (card.parentNode) {
          card.classList.add("removing");
          card.addEventListener("animationend", function () {
            if (card.parentNode) {
              card.parentNode.removeChild(card);
            }
          });
        }
      }, DISMISS_DELAY);
    }
  }

  // --- Event Listeners ---
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
</body>
