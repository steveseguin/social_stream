<script>
(function() {
  // 1. Setup Bridge
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_alert_" + Date.now();

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. State
  var alertContainer = document.getElementById("alert-container");
  var alertTimeouts = {};

  // 3. Helper: Escape HTML
  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // 4. Helper: Create Alert Card
  function createAlertCard(type, name, message, avatarUrl, detail) {
    var card = document.createElement("div");
    card.className = "alert-card type-" + type;
    
    // Avatar
    var img = document.createElement("img");
    img.className = "alert-avatar";
    img.src = avatarUrl || "https://ui-avatars.com/api/?name=" + encodeURIComponent(name) + "&background=random&color=fff";
    img.onerror = function() {
      this.src = "https://ui-avatars.com/api/?name=" + encodeURIComponent(name) + "&background=random&color=fff";
    };
    
    // Content
    var content = document.createElement("div");
    content.className = "alert-content";

    var nameSpan = document.createElement("span");
    nameSpan.className = "alert-name";
    nameSpan.textContent = name;

    var actionSpan = document.createElement("span");
    actionSpan.className = "alert-action";
    actionSpan.textContent = message;

    if (detail) {
      var detailSpan = document.createElement("span");
      detailSpan.className = "alert-action";
      detailSpan.style.opacity = "0.7";
      detailSpan.textContent = detail;
      content.appendChild(nameSpan);
      content.appendChild(actionSpan);
      content.appendChild(detailSpan);
    } else {
      content.appendChild(nameSpan);
      content.appendChild(actionSpan);
    }

    card.appendChild(img);
    card.appendChild(content);

    // Append to container
    alertContainer.appendChild(card);

    // Auto-dismiss after 4 seconds
    var timeoutId = setTimeout(function() {
      card.classList.add("fade-out");
      card.addEventListener("animationend", function() {
        if (card.parentNode) {
          card.parentNode.removeChild(card);
        }
      });
    }, 4000);

    alertTimeouts[card] = timeoutId;
  }

  // 5. Handle Payload
  function handlePayload(data) {
    if (!data) return;
    
    // Determine Type
    var type = "";
    var name = data.chatname || "Unknown User";
    var avatar = data.chatimg || "";
    var msg = "";
    var detail = "";

    // Follower
    if (data.event === "new_follower") {
      type = "follow";
      msg = "just followed!";
    }
    // Subscriber (Resub or New Sub)
    else if (data.event === "subscriber" || data.event === "resub" || (data.membership && data.event !== "subscription_gift" && data.event !== "gifted")) {
      type = "sub";
      msg = "subscribed!";
      if (data.membership) detail = data.membership;
      if (data.subtitle) detail += " • " + data.subtitle;
    }
    // Gifted Sub
    else if (data.event === "subscription_gift" || data.event === "gifted") {
      type = "gift";
      msg = "gifted a sub!";
      if (data.subtitle) detail = data.subtitle;
      if (data.meta && data.meta.gift_count) detail = "Gifted " + data.meta.gift_count + " subs!";
    }
    // Donation
    else if (data.hasDonation) {
      type = "donate";
      msg = "donated " + data.hasDonation;
      if (data.chatmessage) detail = data.chatmessage;
    }
    // Raid
    else if (data.event === "raid") {
      type = "raid";
      msg = "is raiding with";
      var viewers = data.subtitle || (data.meta && data.meta.viewers) || "viewers";
      detail = viewers;
    }

    // Render if valid type
    if (type) {
      createAlertCard(type, name, msg, avatar, detail);
    }
  }

  // 6. Event Listeners
  window.addEventListener("message", function(event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
</body>
