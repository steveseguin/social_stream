<script>
(function () {
  // 1. Setup Bridge
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. Configuration
  var limit = parseInt(params.get("limit")) || 10;
  var chatContainer = document.getElementById("chat-container");
  var chatRows = []; // Keep track of DOM elements to manage limit

  // 3. Helper: Escape HTML to prevent XSS
  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // 4. Render Logic
  function renderChat(data) {
    // Filter: Must be chat, not bot, not event
    if (data.event || data.bot === true) return;
    if (!data.chatname && !data.chatmessage) return;

    // Create Row
    var row = document.createElement("div");
    row.className = "chat-row";

    // Avatar
    var avatar = document.createElement("img");
    avatar.className = "chat-avatar";
    if (data.chatimg) {
      avatar.src = data.chatimg;
      // Fallback if image fails
      avatar.onerror = function() {
        this.style.display = "none";
      };
    } else {
      avatar.style.display = "none";
    }
    row.appendChild(avatar);

    // Content Wrapper
    var content = document.createElement("div");
    content.className = "chat-content";

    // Meta (Platform/Time)
    var meta = document.createElement("div");
    meta.className = "chat-meta";
    if (data.type) {
      meta.textContent = data.type.toUpperCase();
    }
    content.appendChild(meta);

    // Name
    var nameSpan = document.createElement("span");
    nameSpan.className = "chat-name";
    nameSpan.textContent = data.chatname;
    
    // Message
    var msgSpan = document.createElement("span");
    msgSpan.className = "chat-message";
    msgSpan.textContent = data.chatmessage;

    content.appendChild(nameSpan);
    content.appendChild(msgSpan);
    row.appendChild(content);

    // Add to DOM
    chatContainer.appendChild(row);
    chatRows.push(row);

    // Enforce Limit (Remove oldest)
    while (chatRows.length > limit) {
      var oldRow = chatRows.shift();
      if (oldRow && oldRow.parentNode) {
        oldRow.parentNode.removeChild(oldRow);
      }
    }
  }

  // 5. Message Handler
  function handlePayload(data) {
    renderChat(data);
  }

  // Listen for Bridge Messages
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Allow manual testing via console
  window.handleOverlayPayload = handlePayload;

})();
</script>
