<script>
(function () {
  "use strict";

  var platformOrder = ["youtube", "twitch", "kick", "tiktok", "facebook", "rumble", "discord"];
  var currentCounts = {};
  var prevCounts = {};
  var barVisible = false;

  var bar = document.getElementById("viewer-bar");
  var platformsContainer = document.getElementById("platforms-container");
  var totalCountEl = document.getElementById("total-count");

  function getPlatformColor(type) {
    var t = (type || "").toLowerCase();
    if (t === "youtube") return "youtube";
    if (t === "twitch") return "twitch";
    if (t === "kick") return "kick";
    if (t === "tiktok") return "tiktok";
    if (t === "facebook") return "facebook";
    if (t === "rumble") return "rumble";
    if (t === "discord") return "discord";
    return "default";
  }

  function formatNumber(n) {
    if (typeof n !== "number" || isNaN(n)) return "0";
    return Math.max(0, Math.round(n)).toLocaleString();
  }

  function renderPlatforms() {
    platformsContainer.innerHTML = "";

    var allPlatforms = Object.keys(currentCounts);
    if (allPlatforms.length === 0) return;

    // Sort: known platforms first in order, then unknown ones
    var sorted = allPlatforms.slice().sort(function (a, b) {
      var ia = platformOrder.indexOf(a.toLowerCase());
      var ib = platformOrder.indexOf(b.toLowerCase());
      var aKnown = ia !== -1;
      var bKnown = ib !== -1;
      if (aKnown && bKnown) return ia - ib;
      if (aKnown) return -1;
      if (bKnown) return 1;
      return a.localeCompare(b);
    });

    sorted.forEach(function (plat) {
      var count = currentCounts[plat];
      if (count <= 0) return;

      var el = document.createElement("div");
      el.className = "platform-count";
      el.setAttribute("data-platform", plat);

      var colorClass = getPlatformColor(plat);

      var dot = document.createElement("span");
      dot.className = "dot dot-" + colorClass;
      el.appendChild(dot);

      var label = document.createElement("span");
      label.className = "label label-" + colorClass;
      label.textContent = plat;
      el.appendChild(label);

      var countSpan = document.createElement("span");
      countSpan.className = "count";
      countSpan.textContent = formatNumber(count);
      countSpan.setAttribute("data-count-for", plat);
      el.appendChild(countSpan);

      platformsContainer.appendChild(el);
    });
  }

  function updateTotal() {
    var total = 0;
    var keys = Object.keys(currentCounts);
    for (var i = 0; i < keys.length; i++) {
      total += Math.max(0, Math.round(currentCounts[keys[i]]));
    }

    var newTotalStr = formatNumber(total);
    var oldTotalStr = totalCountEl.textContent;

    totalCountEl.textContent = newTotalStr;

    if (newTotalStr !== oldTotalStr) {
      totalCountEl.classList.remove("changed");
      void totalCountEl.offsetWidth; // reflow
      totalCountEl.classList.add("changed");
    }
  }

  function animateCountChange(platform) {
    var countSpan = platformsContainer.querySelector('[data-count-for="' + platform + '"]');
    if (countSpan) {
      countSpan.classList.remove("changed");
      void countSpan.offsetWidth;
      countSpan.classList.add("changed");
    }
  }

  function handlePayload(data) {
    if (!data || !data.event || data.event !== "viewer_updates") return;

    var meta = data.meta;
    if (!meta || typeof meta !== "object") return;

    // Snapshot: replace state entirely
    var newCounts = {};
    var keys = Object.keys(meta);
    for (var i = 0; i < keys.length; i++) {
      var k = keys[i];
      var v = meta[k];
      if (typeof v === "number" && v > 0) {
        newCounts[k] = v;
      }
    }

    // Detect changes
    var allKeys = Object.keys(newCounts);
    for (var j = 0; j < allKeys.length; j++) {
      var pk = allKeys[j];
      var oldVal = prevCounts[pk] || 0;
      var newVal = newCounts[pk];
      if (oldVal !== newVal) {
        animateCountChange(pk);
      }
    }

    prevCounts = {};
    var prevKeys = Object.keys(currentCounts);
    for (var p = 0; p < prevKeys.length; p++) {
      prevCounts[prevKeys[p]] = currentCounts[prevKeys[p]];
    }

    currentCounts = newCounts;

    // Show bar if we have data
    if (allKeys.length > 0 && !barVisible) {
      barVisible = true;
      bar.classList.add("visible");
    }

    renderPlatforms();
    updateTotal();
  }

  // --- VDO.Ninja Bridge ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  window.handleOverlayPayload = handlePayload;

  // --- Test mode: if no session param, show demo bar after 1s ---
  if (!roomID) {
    setTimeout(function () {
      currentCounts = { youtube: 1243, twitch: 567, kick: 89, tiktok: 2100 };
      prevCounts = {};
      barVisible = true;
      bar.classList.add("visible");
      renderPlatforms();
      updateTotal();
    }, 1000);
  }

})();
</script>
</body>
