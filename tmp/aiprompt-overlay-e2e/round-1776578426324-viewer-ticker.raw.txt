<script>
(function () {
  // 1. Setup Bridge
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. State Management
  var currentCounts = {};
  var container = document.getElementById("container");
  var platformsList = document.getElementById("platforms-list");
  var totalCountEl = document.getElementById("total-count");

  // 3. Render Logic
  function renderCounts() {
    var platforms = Object.keys(currentCounts).sort();
    var total = 0;
    
    // Clear existing platform elements
    platformsList.innerHTML = "";

    if (platforms.length === 0) {
      container.classList.remove("visible");
      return;
    }

    container.classList.add("visible");

    platforms.forEach(function (platform) {
      var count = currentCounts[platform];
      total += count;

      var div = document.createElement("div");
      div.className = "platform-count";

      var nameSpan = document.createElement("span");
      nameSpan.className = "platform-name";
      nameSpan.textContent = platform;

      var valSpan = document.createElement("span");
      valSpan.className = "platform-value";
      valSpan.textContent = count.toLocaleString();

      div.appendChild(nameSpan);
      div.appendChild(valSpan);
      platformsList.appendChild(div);
    });

    // Update total
    totalCountEl.textContent = total.toLocaleString();
  }

  // 4. Message Handler
  function handlePayload(data) {
    // We only care about viewer_updates events
    if (data && data.event === "viewer_updates" && data.meta) {
      // Rule 14: meta is a FULL SNAPSHOT. Replace state, don't merge.
      currentCounts = {};
      
      // Iterate over the meta object keys (platforms)
      for (var key in data.meta) {
        if (data.meta.hasOwnProperty(key)) {
          var val = parseInt(data.meta[key], 10);
          if (!isNaN(val)) {
            currentCounts[key] = val;
          }
        }
      }
      
      renderCounts();
    }
  }

  // 5. Event Listeners
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Allow manual testing via console: window.handleOverlayPayload({ event: "viewer_updates", meta: { youtube: 100, twitch: 50 } });
  window.handleOverlayPayload = handlePayload;

})();
</script>
