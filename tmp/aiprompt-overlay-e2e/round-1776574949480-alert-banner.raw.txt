<script>
(function () {
  // --- VDO.Ninja Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // --- State & Helpers ---
  var container = document.getElementById("alert-container");
  var DISMISS_DELAY = 4000; // 4 seconds

  function escapeHtml(text) {
    if (!text) return "";
    return text
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#039;");
  }

  function getAvatarUrl(data) {
    return data.chatimg || "data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100' fill='%23555'%3E%3Ccircle cx='50' cy='50' r='50'/%3E%3C/svg%3E";
  }

  // --- Alert Rendering Logic ---

  function createAlertCard(type, data) {
    var card = document.createElement("div");
    card.className = "alert-card type-" + type;

    // Avatar
    var img = document.createElement("img");
    img.className = "alert-avatar";
    img.src = getAvatarUrl(data);
    img.onerror = function() { this.src = "data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100' fill='%23555'%3E%3Ccircle cx='50' cy='50' r='50'/%3E%3C/svg%3E"; };
    card.appendChild(img);

    // Content
    var content = document.createElement("div");
    content.className = "alert-content";

    var header = document.createElement("div");
    header.className = "alert-header";

    var nameSpan = document.createElement("span");
    nameSpan.className = "alert-name";
    nameSpan.textContent = data.chatname || "Unknown";
    header.appendChild(nameSpan);

    var actionSpan = document.createElement("span");
    actionSpan.className = "alert-action";
    content.appendChild(header);

    var detailSpan = document.createElement("div");
    detailSpan.className = "alert-detail";
    content.appendChild(detailSpan);

    card.appendChild(content);

    // Type specific text
    switch (type) {
      case "follow":
        actionSpan.textContent = "followed!";
        break;
      case "sub":
        actionSpan.textContent = "subscribed!";
        if (data.membership) {
          detailSpan.textContent = escapeHtml(data.membership);
        }
        break;
      case "gift":
        actionSpan.textContent = "gifted subscriptions!";
        var giftCount = data.subtitle || (data.meta && data.meta.gift_count) || "";
        if (giftCount) {
          detailSpan.textContent = "Gifted: " + escapeHtml(String(giftCount));
        }
        break;
      case "donate":
        actionSpan.textContent = "donated " + escapeHtml(data.hasDonation);
        if (data.chatmessage) {
          detailSpan.textContent = '"' + escapeHtml(data.chatmessage) + '"';
        }
        break;
      case "raid":
        actionSpan.textContent = "is raiding with";
        var viewerCount = data.subtitle || (data.meta && (data.meta.viewers || data.meta.count)) || "";
        if (viewerCount) {
          detailSpan.textContent = escapeHtml(String(viewerCount)) + " viewers";
        } else {
          detailSpan.textContent = "viewers";
        }
        break;
    }

    // Progress bar
    var progress = document.createElement("div");
    progress.className = "alert-progress";
    card.appendChild(progress);

    return card;
  }

  function dismissAlert(card) {
    card.classList.add("fade-out");
    setTimeout(function() {
      if (card.parentNode) {
        card.parentNode.removeChild(card);
      }
    }, 500); // Wait for fade out animation
  }

  function showAlert(type, data) {
    var card = createAlertCard(type, data);
    container.appendChild(card);
    setTimeout(function() {
      dismissAlert(card);
    }, DISMISS_DELAY);
  }

  // --- Message Router ---

  function handlePayload(data) {
    if (!data) return;

    // 1. Followers
    if (data.event === "new_follower") {
      showAlert("follow", data);
      return;
    }

    // 2. Subscribers (explicit event or membership present)
    if (data.event === "subscriber" || data.event === "resub") {
      showAlert("sub", data);
      return;
    }
    // Fallback: if no specific sub event but has membership string
    if (!data.event && data.membership && !data.hasDonation && !data.event) {
       // Only treat as sub if it looks like a sub event (usually triggered by SSN on join)
       // We rely on explicit events mostly, but this catches edge cases
    }

    // 3. Gifted Subs
    if (data.event === "subscription_gift" || data.event === "gifted") {
      showAlert("gift", data);
      return;
    }

    // 4. Donations
    if (data.hasDonation) {
      showAlert("donate", data);
      return;
    }

    // 5. Raids
    if (data.event === "raid") {
      showAlert("raid", data);
      return;
    }
  }

  // --- Event Listeners ---

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;

})();
</script>
