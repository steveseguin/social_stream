<script>
(function() {
  // VDO.Ninja Bridge Setup
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_alerts_" + Date.now();
  
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;

  var container = document.getElementById("alert-container");
  var alertTimeout = 4000; // 4 seconds

  function handlePayload(data) {
    if (!data) return;

    var type = data.type || "";
    var event = data.event || "";
    var name = data.chatname || "Anonymous";
    var msg = data.chatmessage || "";
    var img = data.chatimg || "";
    var donation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";

    var alertType = null;

    // 1. New Follower
    if (event === "new_follower") {
      alertType = "follow";
      msg = msg || "Just followed!";
    }
    // 2. Subscriber (Resub)
    else if (event === "subscriber" || (membership && event !== "subscription_gift" && event !== "gifted")) {
      alertType = "sub";
      msg = msg || ("Subscribed! " + (subtitle || ""));
    }
    // 3. Gifted Sub
    else if (event === "subscription_gift" || event === "gifted") {
      alertType = "gift";
      msg = msg || ("Gifted subs! " + (subtitle || ""));
    }
    // 4. Donation
    else if (donation) {
      alertType = "donate";
      msg = msg || "Thanks for the support!";
    }
    // 5. Raid
    else if (event === "raid") {
      alertType = "raid";
      msg = msg || ("Raiding with " + (data.subtitle || "viewers"));
    }

    if (alertType) {
      showAlert(alertType, name, msg, img, donation, membership);
    }
  }

  function showAlert(type, name, msg, img, donation, membership) {
    var card = document.createElement("div");
    card.className = "alert-card type-" + type;

    // Icon
    var icon = document.createElement("div");
    icon.className = "user-icon";
    if (img) icon.style.backgroundImage = "url('" + img + "')";
    
    // Text
    var textGroup = document.createElement("div");
    textGroup.className = "text-group";

    var userName = document.createElement("div");
    userName.className = "user-name";
    userName.textContent = name;

    var messageText = document.createElement("div");
    messageText.className = "alert-message";
    
    // Construct message based on type
    if (type === "donate") {
      messageText.textContent = donation + " — " + msg;
    } else if (type === "sub" && membership) {
      messageText.textContent = membership + " — " + msg;
    } else {
      messageText.textContent = msg;
    }

    textGroup.appendChild(userName);
    textGroup.appendChild(messageText);

    // Badge
    var badge = document.createElement("div");
    badge.className = "alert-badge";
    if (type === "follow") badge.textContent = "New Follower";
    else if (type === "sub") badge.textContent = "Subscriber";
    else if (type === "gift") badge.textContent = "Gifted Sub";
    else if (type === "donate") badge.textContent = "Donation";
    else if (type === "raid") badge.textContent = "RAID";

    card.appendChild(icon);
    card.appendChild(textGroup);
    card.appendChild(badge);

    container.appendChild(card);

    // Auto dismiss
    setTimeout(function() {
      card.classList.add("removing");
      card.addEventListener("animationend", function() {
        if (card.parentNode) {
          card.parentNode.removeChild(card);
        }
      });
    }, alertTimeout);
  }
})();
</script>
