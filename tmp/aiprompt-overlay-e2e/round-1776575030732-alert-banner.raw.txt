<script>
(function () {
  "use strict";

  var alertContainer = document.getElementById("alert-container");
  var alertStack = [];
  var MAX_VISIBLE = 3;
  var DISMISS_DELAY = 4000;

  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  function escapeHTML(str) {
    if (!str) return "";
    var div = document.createElement("div");
    div.appendChild(document.createTextNode(str));
    return div.innerHTML;
  }

  function handlePayload(data) {
    if (!data) return;

    var alertType = classifyAlert(data);
    if (!alertType) return;

    showAlert(data, alertType);
  }

  function classifyAlert(data) {
    if (data.event === "new_follower") return "new_follower";
    if (data.event === "subscriber") return "subscriber";
    if (data.event === "subscription_gift") return "subscription_gift";
    if (data.event === "gifted") return "gifted";
    if (data.event === "raid") return "raid";
    if (data.hasDonation) return "donation";
    if (data.membership && !data.event) return "subscriber";
    return null;
  }

  function getAlertLabel(type) {
    switch (type) {
      case "new_follower": return "FOLLOWER";
      case "subscriber": return "SUBSCRIBER";
      case "subscription_gift": return "GIFTED SUB";
      case "gifted": return "GIFTED";
      case "donation": return "DONATION";
      case "raid": return "RAID";
      default: return "ALERT";
    }
  }

  function getInitials(name) {
    if (!name) return "?";
    var parts = name.split(/\s+/);
    if (parts.length >= 2) return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase();
    return name.substring(0, 2).toUpperCase();
  }

  function buildAlertCard(data, alertType) {
    var card = document.createElement("div");
    card.className = "alert-card";
    card.setAttribute("data-type", alertType);

    var chatname = data.chatname || "Anonymous";
    var chatmessage = data.chatmessage || "";
    var chatimg = data.chatimg || "";
    var hasDonation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";
    var meta = data.meta || {};

    var label = getAlertLabel(alertType);
    var messageText = "";
    var detailText = "";

    switch (alertType) {
      case "new_follower":
        messageText = "followed the channel";
        break;
      case "subscriber":
        messageText = "subscribed";
        if (membership) {
          detailText = escapeHTML(membership);
        }
        break;
      case "subscription_gift":
      case "gifted":
        var giftCount = subtitle;
        if (meta && meta.gift_count) {
          giftCount = meta.gift_count;
        }
        messageText = "gifted " + (giftCount || "a") + " sub";
        break;
      case "donation":
        messageText = "donated " + escapeHTML(hasDonation);
        if (chatmessage) {
          detailText = escapeHTML(chatmessage);
        }
        break;
      case "raid":
        var viewerCount = subtitle;
        if (meta && meta.viewers) {
          viewerCount = meta.viewers;
        }
        messageText = "is raiding with " + viewerCount + " viewers";
        break;
    }

    if (data.type) {
      if (!detailText) {
        detailText = escapeHTML(data.type);
      } else {
        detailText = detailText + " • " + escapeHTML(data.type);
      }
    }

    var avatarHTML = "";
    if (chatimg) {
      var img = document.createElement("img");
      img.className = "alert-avatar";
      img.src = chatimg;
      img.alt = escapeHTML(chatname);
      img.onerror = function () {
        this.style.display = "none";
        var placeholder = document.createElement("div");
        placeholder.className = "alert-avatar-placeholder";
        placeholder.textContent = getInitials(chatname);
        this.parentNode.insertBefore(placeholder, this.nextSibling);
      };
      avatarHTML = img.outerHTML;
      card.appendChild(img);
    } else {
      var placeholder = document.createElement("div");
      placeholder.className = "alert-avatar-placeholder";
      placeholder.textContent = getInitials(chatname);
      card.appendChild(placeholder);
    }

    var badge = document.createElement("span");
    badge.className = "alert-badge";
    badge.textContent = label;
    card.appendChild(badge);

    var content = document.createElement("div");
    content.className = "alert-content";

    var nameEl = document.createElement("div");
    nameEl.className = "alert-name";
    nameEl.textContent = chatname;
    content.appendChild(nameEl);

    var messageEl = document.createElement("div");
    messageEl.className = "alert-message";
    messageEl.textContent = messageText;
    content.appendChild(messageEl);

    if (detailText) {
      var detailEl = document.createElement("div");
      detailEl.className = "alert-detail";
      detailEl.textContent = detailText;
      content.appendChild(detailEl);
    }

    card.appendChild(content);

    if (data.contentimg) {
      var contentImg = document.createElement("img");
      contentImg.className = "alert-image";
      contentImg.src = data.contentimg;
      contentImg.alt = "Alert image";
      contentImg.onerror = function () {
        this.style.display = "none";
      };
      card.appendChild(contentImg);
    }

    return card;
  }

  function showAlert(data, alertType) {
    var card = buildAlertCard(data, alertType);
    alertContainer.appendChild(card);

    var dismissTimer = setTimeout(function () {
      dismissCard(card);
    }, DISMISS_DELAY);

    var alertEntry = { card: card, timer: dismissTimer };
    alertStack.push(alertEntry);

    enforceMaxVisible();
  }

  function dismissCard(card) {
    card.classList.add("dismissing");
    var entry = alertStack.find(function (a) { return a.card === card; });
    if (entry) {
      clearTimeout(entry.timer);
    }
    setTimeout(function () {
      if (card.parentNode) {
        card.parentNode.removeChild(card);
      }
      alertStack = alertStack.filter(function (a) { return a.card !== card; });
    }, 300);
  }

  function enforceMaxVisible() {
    while (alertStack.length > MAX_VISIBLE) {
      var oldest = alertStack.shift();
      clearTimeout(oldest.timer);
      dismissCard(oldest.card);
    }
  }

  window.handleOverlayPayload = handlePayload;
  window.showTestAlert = function () {
    var testPayloads = [
      { chatname: "FollowerFan", event: "new_follower", type: "twitch", chatimg: "" },
      { chatname: "SubSally", event: "subscriber", type: "youtube", membership: "Tier 2", chatimg: "" },
      { chatname: "GiftyGus", event: "subscription_gift", type: "kick", subtitle: "5", chatimg: "" },
      { chatname: "DonorDana", hasDonation: "$25.00", chatmessage: "Keep it up!", type: "youtube", chatimg: "" },
      { chatname: "RaiderRita", event: "raid", type: "twitch", subtitle: "42", chatimg: "" }
    ];
    var random = testPayloads[Math.floor(Math.random() * testPayloads.length)];
    handlePayload(random);
  };
})();
</script>
