<script>
(function () {
  // 1. Setup Bridge
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_viewers_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. State Management
  var currentCounts = {}; // { youtube: 100, twitch: 50 }
  var container = document.getElementById("container");

  // 3. Render Logic
  function renderCounts() {
    container.innerHTML = ""; // Clear current
    
    var platforms = Object.keys(currentCounts);
    var total = 0;

    // If no data, hide container or show 0
    if (platforms.length === 0) {
      // Optional: Show a placeholder or hide
      // container.style.display = 'none'; 
      return;
    }

    // Render individual platforms
    platforms.forEach(function(p) {
      var count = currentCounts[p];
      total += count;

      var statDiv = document.createElement("div");
      statDiv.className = "platform-stat";

      var nameSpan = document.createElement("span");
      nameSpan.className = "platform-name";
      nameSpan.textContent = p;

      var countSpan = document.createElement("span");
      countSpan.className = "platform-count";
      countSpan.textContent = count.toLocaleString();

      statDiv.appendChild(nameSpan);
      statDiv.appendChild(countSpan);
      container.appendChild(statDiv);
    });

    // Render Total
    var totalDiv = document.createElement("div");
    totalDiv.className = "total-stat";

    var totalName = document.createElement("span");
    totalName.className = "platform-name";
    totalName.textContent = "Total";

    var totalCountSpan = document.createElement("span");
    totalCountSpan.className = "total-count";
    totalCountSpan.textContent = total.toLocaleString();

    totalDiv.appendChild(totalName);
    totalDiv.appendChild(totalCountSpan);
    container.appendChild(totalDiv);
  }

  // 4. Message Handler
  function handlePayload(data) {
    // We only care about viewer_updates snapshots
    if (data && data.event === "viewer_updates" && data.meta) {
      var newMeta = data.meta;
      
      // Check if data actually changed to trigger animations
      var changed = false;
      var newKeys = Object.keys(newMeta);
      var oldKeys = Object.keys(currentCounts);
      
      if (newKeys.length !== oldKeys.length) changed = true;
      else {
        for (var i = 0; i < newKeys.length; i++) {
          if (newMeta[newKeys[i]] !== currentCounts[newKeys[i]]) {
            changed = true;
            break;
          }
        }
      }

      currentCounts = newMeta;
      renderCounts();

      // Simple animation trigger: re-add class to counts if changed
      if (changed) {
        var countEls = container.querySelectorAll(".platform-count, .total-count");
        countEls.forEach(function(el) {
          el.classList.remove("pop");
          void el.offsetWidth; // trigger reflow
          el.classList.add("pop");
        });
      }
    }
  }

  // 5. Event Listeners
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // Allow manual testing via console
  window.handleOverlayPayload = handlePayload;

})();
</script>
