<script>
(function () {
  // --- Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // --- State & Config ---
  var container = document.getElementById("alert-container");
  var MAX_ALERTS = 3; // Keep stack small
  var DISMISS_TIME = 4000; // 4 seconds

  // --- Helpers ---
  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  function getAvatarUrl(data) {
    return data.chatimg || "data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100'%3E%3Crect width='100' height='100' fill='%23333'/%3E%3Ctext x='50' y='50' font-size='40' text-anchor='middle' dy='.3em' fill='white'%3E%3F%3C/text%3E%3C/svg%3E";
  }

  // --- Alert Logic ---
  function handlePayload(data) {
    if (!data) return;

    var type = data.type || "";
    var eventName = data.event || "";
    var hasDonation = data.hasDonation;
    var membership = data.membership;
    var name = data.chatname || "Anonymous";
    var message = data.chatmessage || "";
    var subtitle = data.subtitle || "";
    var img = getAvatarUrl(data);

    // Determine if this is an alertable event
    var isAlert = false;
    var alertClass = "";
    var actionText = "";
    var subText = "";

    // 1. New Follower
    if (eventName === "new_follower") {
      isAlert = true;
      alertClass = "type-new_follower";
      actionText = "followed";
    }
    // 2. Subscriber (Resub or New Sub)
    else if (eventName === "subscriber" || (eventName === "resub" && !eventName === "subscription_gift")) {
      isAlert = true;
      alertClass = "type-subscriber";
      actionText = "subscribed";
      if (membership) subText = membership;
    }
    // 3. Gifted Sub
    else if (eventName === "subscription_gift" || eventName === "gifted") {
      isAlert = true;
      alertClass = "type-subscription_gift";
      actionText = "gifted a sub";
      if (subtitle) subText = subtitle; // e.g., "5 subs"
    }
    // 4. Donation
    else if (hasDonation) {
      isAlert = true;
      alertClass = "type-donation";
      actionText = "donated " + escapeHtml(hasDonation);
      if (message) subText = '"' + escapeHtml(message) + '"';
    }
    // 5. Raid
    else if (eventName === "raid") {
      isAlert = true;
      alertClass = "type-raid";
      actionText = "is raiding with";
      // Raid viewer count often in subtitle or meta
      var viewers = subtitle || (data.meta && data.meta.viewers) || "";
      subText = viewers + " viewers";
    }

    if (isAlert) {
      createAlertCard(name, actionText, subText, img, alertClass);
    }
  }

  function createAlertCard(name, action, sub, img, cssClass) {
    // Remove oldest if at max
    if (container.children.length >= MAX_ALERTS) {
      var oldest = container.children[0];
      oldest.classList.add("fade-out");
      setTimeout(function() {
        if (oldest.parentNode) oldest.parentNode.removeChild(oldest);
      }, 500); // Match CSS animation duration
    }

    var card = document.createElement("div");
    card.className = "alert-card " + cssClass;

    // Avatar
    var avatar = document.createElement("img");
    avatar.className = "user-avatar";
    avatar.src = img;
    // Fallback if image fails
    avatar.onerror = function() { this.src = img; }; // Reset to default or keep broken

    // Content
    var content = document.createElement("div");
    content.className = "alert-content";

    // Header: Name + Action
    var header = document.createElement("div");
    header.className = "alert-header";
    
    var nameSpan = document.createElement("span");
    nameSpan.className = "user-name";
    nameSpan.textContent = name;

    var actionSpan = document.createElement("span");
    actionSpan.className = "alert-action";
    actionSpan.textContent = " " + action;

    header.appendChild(nameSpan);
    header.appendChild(actionSpan);

    // Subtext
    if (sub) {
      var subSpan = document.createElement("div");
      subSpan.className = "alert-sub";
      subSpan.textContent = sub;
      content.appendChild(header);
      content.appendChild(subSpan);
    } else {
      content.appendChild(header);
    }

    card.appendChild(avatar);
    card.appendChild(content);
    container.appendChild(card);

    // Auto dismiss
    setTimeout(function() {
      card.classList.add("fade-out");
      setTimeout(function() {
        if (card.parentNode) card.parentNode.removeChild(card);
      }, 500);
    }, DISMISS_TIME);
  }

  // --- Event Listeners ---
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
