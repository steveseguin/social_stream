<script>
(function () {
  // --- Configuration ---
  var limit = 10;
  var params = new URLSearchParams(location.search);
  if (params.has("limit")) {
    var parsedLimit = parseInt(params.get("limit"), 10);
    if (!isNaN(parsedLimit) && parsedLimit > 0) {
      limit = parsedLimit;
    }
  }

  var chatContainer = document.getElementById("chat-container");
  var messages = []; // Array to store message objects

  // --- Bridge Setup ---
  var bridge = document.getElementById("ssn_bridge");
  if (bridge) {
    var roomID = params.get("session") || "";
    var password = params.get("password") || "false";
    var label = params.get("label") || "custom_overlay_chat_" + Date.now();
    
    if (roomID) {
      bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
        encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
        "&label=" + encodeURIComponent(label) +
        "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
    }
  }

  // --- Message Handling ---
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  function handlePayload(data) {
    // Filter: Must be chat (no event) and not a bot
    if (data.event || data.bot) return;
    if (!data.chatname && !data.chatmessage) return;

    var msg = {
      id: data.id || Date.now(),
      chatname: data.chatname || "Anonymous",
      chatmessage: data.chatmessage || "",
      chatimg: data.chatimg || "",
      nameColor: data.nameColor || null
    };

    messages.push(msg);

    // Enforce limit
    if (messages.length > limit) {
      messages.shift();
    }

    renderChat();
  }

  function renderChat() {
    // Clear current container
    chatContainer.innerHTML = "";

    // Render messages
    for (var i = 0; i < messages.length; i++) {
      var msg = messages[i];
      var row = document.createElement("div");
      row.className = "chat-row";

      // Avatar
      if (msg.chatimg) {
        var img = document.createElement("img");
        img.className = "avatar";
        img.src = msg.chatimg;
        img.alt = msg.chatname + " avatar";
        img.onerror = function() {
          this.style.display = 'none';
          var placeholder = document.createElement('div');
          placeholder.className = 'avatar-placeholder';
          placeholder.textContent = msg.chatname.charAt(0).toUpperCase();
          row.insertBefore(placeholder, row.firstChild);
        };
        row.appendChild(img);
      } else {
        var placeholder = document.createElement("div");
        placeholder.className = "avatar-placeholder";
        placeholder.textContent = msg.chatname.charAt(0).toUpperCase();
        row.appendChild(placeholder);
      }

      // Content
      var content = document.createElement("div");
      content.className = "message-content";

      var nameSpan = document.createElement("span");
      nameSpan.className = "username";
      nameSpan.textContent = msg.chatname;
      if (msg.nameColor) {
        nameSpan.style.color = msg.nameColor;
      }

      var textSpan = document.createElement("span");
      textSpan.className = "text";
      textSpan.textContent = msg.chatmessage;

      content.appendChild(nameSpan);
      content.appendChild(textSpan);
      row.appendChild(content);

      chatContainer.appendChild(row);
    }
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
