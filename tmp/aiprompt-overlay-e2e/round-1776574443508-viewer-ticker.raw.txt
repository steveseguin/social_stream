  /* ---- Render ---- */
  function render() {
    var keys = Object.keys(viewerState);
    var total = 0;

    /* Clear existing platform elements */
    platformsEl.innerHTML = "";

    for (var i = 0; i < keys.length; i++) {
      var key = keys[i];
      var count = viewerState[key];
      if (typeof count !== "number" || count < 0) continue;
      total += count;

      var block = document.createElement("div");
      block.className = "platform-count";

      var nameSpan = document.createElement("span");
      nameSpan.className = "platform-name";
      nameSpan.textContent = platformNames[key] || key;

      var numSpan = document.createElement("span");
      numSpan.className = "platform-num";
      numSpan.textContent = count.toLocaleString();

      block.appendChild(nameSpan);
      block.appendChild(numSpan);
      platformsEl.appendChild(block);
    }

    /* Total */
    var newTotal = total;
    totalEl.textContent = newTotal.toLocaleString();

    /* Pulse if total changed */
    if (newTotal !== prevTotal) {
      totalEl.classList.remove("pulse");
      /* Force reflow to restart animation */
      void totalEl.offsetWidth;
      totalEl.classList.add("pulse");
      prevTotal = newTotal;
    }

    /* Show bar if we have data */
    if (keys.length > 0) {
      bar.classList.add("visible");
    }
  }

  /* ---- Handle payload ---- */
  function handlePayload(data) {
    if (!data) return;

    /* Only process viewer_updates snapshots */
    if (data.event === "viewer_updates" && data.meta) {
      var meta = data.meta;
      /* meta is a full snapshot — replace state entirely */
      viewerState = {};
      var keys = Object.keys(meta);
      for (var i = 0; i < keys.length; i++) {
        var k = keys[i];
        var v = meta[k];
        /* Support both number and object with count property */
        if (typeof v === "number") {
          viewerState[k] = v;
        } else if (v && typeof v === "object") {
          viewerState[k] = typeof v.count === "number" ? v.count : 0;
        }
      }
      render();
    }
  }

  /* ---- Bridge setup ---- */
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
