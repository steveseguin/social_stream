<script>
(function () {
  "use strict";

  /* ---- Config ---- */
  var params = new URLSearchParams(location.search);
  var limit = parseInt(params.get("limit"), 10) || 10;
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_chat_" + Date.now();

  var container = document.getElementById("chat-container");
  var bridge = document.getElementById("ssn_bridge");
  var rows = [];

  /* ---- Platform color map ---- */
  var platformColors = {
    youtube: "#ff0000",
    twitch: "#9146ff",
    kick: "#53fc18",
    tiktok: "#fe2c55",
    facebook: "#1877f2",
    rumble: "#8b0000",
    discord: "#5865f2",
    whatnot: "#ff6b35"
  };

  /* ---- Bridge setup ---- */
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  window.handleOverlayPayload = handlePayload;

  /* ---- Core handler ---- */
  function handlePayload(data) {
    if (!data || typeof data !== "object") return;

    /* Filter: skip bots */
    if (data.bot === true) return;

    /* Filter: skip system events (only show plain chat) */
    if (data.event) return;

    /* Filter: must have a chatname */
    if (!data.chatname) return;

    addChatRow(data);
  }

  /* ---- Render a single chat row ---- */
  function addChatRow(data) {
    var row = document.createElement("div");
    row.className = "chat-row";

    /* Avatar */
    var avatarHTML = "";
    if (data.chatimg) {
      avatarHTML = '<img class="avatar" src="' + escapeAttr(data.chatimg) + '" alt="" onerror="this.outerHTML=\'' + avatarPlaceholder(data.chatname) + '\'">';
    } else {
      avatarHTML = avatarPlaceholder(data.chatname);
    }

    /* Username */
    var nameSpan = '<span class="chat-name">' + escapeHTML(data.chatname) + '</span>';

    /* Platform badge */
    var platformBadge = "";
    if (data.type) {
      var pColor = platformColors[data.type] || "#888";
      platformBadge = '<span class="chat-platform" style="background:' + pColor + '22;color:' + pColor + '">' + escapeHTML(data.type) + '</span>';
    }

    /* Message text */
    var msgText = data.chatmessage ? escapeHTML(data.chatmessage) : "";

    row.innerHTML =
      avatarHTML +
      '<div class="chat-content">' +
        '<div class="chat-meta">' + nameSpan + platformBadge + '</div>' +
        (msgText ? '<div class="chat-message">' + msgText + '</div>' : "") +
      '</div>';

    container.appendChild(row);
    rows.push(row);

    /* Enforce limit — fade out oldest */
    while (rows.length > limit) {
      var oldRow = rows.shift();
      oldRow.classList.add("fade-out");
      setTimeout(function () {
        if (oldRow.parentNode) oldRow.parentNode.removeChild(oldRow);
      }, 420);
    }
  }

  /* ---- Helpers ---- */
  function escapeHTML(str) {
    var div = document.createElement("div");
    div.appendChild(document.createTextNode(str));
    return div.innerHTML;
  }

  function escapeAttr(str) {
    return str.replace(/&/g, "&amp;").replace(/"/g, "&quot;").replace(/'/g, "&#39;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
  }

  function avatarPlaceholder(name) {
    var initial = name.charAt(0).toUpperCase();
    return '<div class="avatar-placeholder">' + escapeHTML(initial) + '</div>';
  }

})();
</script>
