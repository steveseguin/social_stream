<script>
(function () {
  "use strict";

  /* ---- Config from URL params ---- */
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "viewer_bar_" + Date.now();
  var hidetype = params.get("hidetype") || "";
  var onlytype = params.get("onlytype") || "";

  var hiddenTypes = hidetype.split(",").map(function (t) { return t.trim().toLowerCase(); });
  var allowedTypes = onlytype.split(",").map(function (t) { return t.trim().toLowerCase(); });

  /* ---- State ---- */
  var viewerState = {};
  var prevCounts = {};

  /* ---- DOM refs ---- */
  var bar = document.getElementById("viewer-bar");
  var segmentsEl = document.getElementById("viewer-segments");
  var totalCountEl = document.getElementById("viewer-total-count");

  /* ---- Helpers ---- */
  function escapeText(str) {
    var div = document.createElement("div");
    div.textContent = str;
    return div.innerHTML;
  }

  function formatNumber(n) {
    return Number(n || 0).toLocaleString("en-US");
  }

  function isHidden(type) {
    if (!type) return false;
    var t = type.toLowerCase();
    for (var i = 0; i < hiddenTypes.length; i++) {
      if (t === hiddenTypes[i]) return true;
    }
    return false;
  }

  function isAllowed(type) {
    if (!onlytype) return true;
    if (!type) return false;
    var t = type.toLowerCase();
    for (var i = 0; i < allowedTypes.length; i++) {
      if (t === allowedTypes[i]) return true;
    }
    return false;
  }

  function getPlatformClass(type) {
    if (!type) return "platform-others";
    var t = type.toLowerCase();
    return "platform-" + t;
  }

  /* ---- Render ---- */
  function render() {
    var keys = Object.keys(viewerState);
    var total = 0;

    segmentsEl.innerHTML = "";

    for (var i = 0; i < keys.length; i++) {
      var key = keys[i];
      var count = viewerState[key];
      var type = key;

      if (isHidden(type) || !isAllowed(type)) continue;
      if (count <= 0) continue;

      total += count;

      var seg = document.createElement("div");
      seg.className = "viewer-segment " + getPlatformClass(type);

      var platformSpan = document.createElement("span");
      platformSpan.className = "viewer-platform";
      platformSpan.textContent = escapeText(type);

      var countSpan = document.createElement("span");
      countSpan.className = "viewer-count";
      countSpan.textContent = formatNumber(count);

      seg.appendChild(platformSpan);
      seg.appendChild(countSpan);
      segmentsEl.appendChild(seg);
    }

    /* Animate total change */
    var newTotalStr = formatNumber(total);
    if (totalCountEl.textContent !== newTotalStr) {
      totalCountEl.textContent = newTotalStr;
      totalCountEl.classList.remove("pulse");
      void totalCountEl.offsetWidth;
      totalCountEl.classList.add("pulse");
    }

    /* Show/hide bar */
    if (total > 0) {
      bar.classList.add("visible");
    } else {
      bar.classList.remove("visible");
    }
  }

  /* ---- Payload handler ---- */
  function handlePayload(data) {
    if (!data) return;

    /* Handle viewer_updates snapshot */
    if (data.event === "viewer_updates" && data.meta && typeof data.meta === "object") {
      var newMeta = data.meta;
      var changed = false;

      /* Replace entire state (snapshot semantics) */
      viewerState = {};
      for (var key in newMeta) {
        if (Object.prototype.hasOwnProperty.call(newMeta, key)) {
          var val = newMeta[key];
          var num = parseInt(val, 10);
          if (!isNaN(num) && num > 0) {
            viewerState[key] = num;
          }
        }
      }

      render();
      return;
    }

    /* Handle individual viewer_update events */
    if (data.event === "viewer_update" && data.meta) {
      var type = data.type || "unknown";
      var delta = parseInt(data.meta, 10);
      if (isNaN(delta)) return;

      if (!viewerState[type]) viewerState[type] = 0;
      viewerState[type] += delta;
      if (viewerState[type] < 0) viewerState[type] = 0;

      render();
      return;
    }
  }

  /* ---- Bridge setup ---- */
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  /* Allow manual test payloads */
  window.handleOverlayPayload = handlePayload;
  window.handlePayload = handlePayload;

  /* Initial render */
  render();

})();
</script>
