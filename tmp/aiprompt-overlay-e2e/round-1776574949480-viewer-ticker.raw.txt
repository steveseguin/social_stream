<script>
(function () {
  /* ---- VDO.Ninja Bridge ---- */
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  /* ---- State ---- */
  var platformCounts = {};
  var container = document.getElementById("container");
  var platformOrder = ["youtube", "twitch", "kick", "tiktok", "facebook", "rumble", "discord"];

  /* ---- Helpers ---- */
  function esc(str) {
    var d = document.createElement("div");
    d.textContent = str;
    return d.innerHTML;
  }

  function formatNumber(n) {
    if (typeof n !== "number" || isNaN(n)) return "0";
    return n.toLocaleString("en-US");
  }

  function getTotal() {
    var total = 0;
    var keys = Object.keys(platformCounts);
    for (var i = 0; i < keys.length; i++) {
      total += platformCounts[keys[i]];
    }
    return total;
  }

  /* ---- Render ---- */
  function render() {
    container.innerHTML = "";

    var platforms = Object.keys(platformCounts);
    var hasData = platforms.length > 0;

    for (var i = 0; i < platforms.length; i++) {
      var p = platforms[i];
      var count = platformCounts[p];
      if (count <= 0) continue;

      var badge = document.createElement("div");
      badge.className = "platform-badge";
      badge.setAttribute("data-platform", p);

      var nameSpan = document.createElement("span");
      nameSpan.className = "platform-name";
      nameSpan.textContent = p;

      var countSpan = document.createElement("span");
      countSpan.className = "platform-count";
      countSpan.textContent = formatNumber(count);

      badge.appendChild(nameSpan);
      badge.appendChild(countSpan);
      container.appendChild(badge);

      if (i < platforms.length - 1) {
        var sep = document.createElement("div");
        sep.className = "separator";
        container.appendChild(sep);
      }
    }

    /* Total */
    var total = getTotal();
    if (total > 0) {
      var totalBadge = document.createElement("div");
      totalBadge.className = "total-badge";

      var totalName = document.createElement("span");
      totalName.className = "platform-name";
      totalName.textContent = "Total";

      var totalCount = document.createElement("span");
      totalCount.className = "platform-count";
      totalCount.textContent = formatNumber(total);

      totalBadge.appendChild(totalName);
      totalBadge.appendChild(totalCount);
      container.appendChild(totalBadge);
    }
  }

  /* ---- Message Handler ---- */
  function handlePayload(data) {
    if (!data) return;

    /* Handle viewer_updates snapshot */
    if (data.event === "viewer_updates" && data.meta && typeof data.meta === "object") {
      platformCounts = {};
      var keys = Object.keys(data.meta);
      for (var i = 0; i < keys.length; i++) {
        var k = keys[i];
        var v = data.meta[k];
        if (typeof v === "number" && v > 0) {
          platformCounts[k] = v;
        }
      }
      render();
      return;
    }

    /* Also accept meta as a plain number (legacy) */
    if (data.event === "viewer_updates" && typeof data.meta === "number" && data.meta > 0) {
      platformCounts["total"] = data.meta;
      render();
      return;
    }
  }

  /* Listen for messages from bridge */
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  /* Allow manual test payloads */
  window.handleOverlayPayload = handlePayload;
  window.handleViewerUpdates = function (meta) {
    handlePayload({ event: "viewer_updates", meta: meta });
  };

  /* Initial render */
  render();
})();
</script>
