<script>
(function () {
  // --- Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // --- Alert Logic ---
  var container = document.getElementById("alert-container");
  var MAX_ALERTS = 5;

  function handlePayload(data) {
    if (!data) return;

    var type = data.type;
    var event = data.event;
    var name = data.chatname || "Anonymous";
    var msg = data.chatmessage || "";
    var donation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";
    var img = data.chatimg || "";
    var color = data.nameColor || "#ffffff";

    // Determine if this is an alert we should show
    var alertType = null;
    var title = "";
    var detail = "";
    var icon = "";

    if (event === "new_follower") {
      alertType = "new_follower";
      title = name + " followed!";
      detail = "Welcome to the stream!";
      icon = "★";
    } 
    else if (event === "subscriber" || (data.membership && event !== "subscription_gift" && event !== "gifted")) {
      alertType = "subscriber";
      title = name + " subscribed!";
      detail = membership || "Thank you for the support!";
      icon = "♥";
    }
    else if (event === "subscription_gift" || event === "gifted") {
      alertType = "subscription_gift";
      // Try to get count from subtitle or meta
      var count = subtitle;
      if (!count && data.meta && data.meta.gift_count) count = data.meta.gift_count;
      title = name + " gifted " + (count || "subs");
      detail = "Thank you for the generosity!";
      icon = "🎁";
    }
    else if (data.hasDonation) {
      alertType = "donation";
      title = name + " donated " + donation;
      detail = msg || "Thank you!";
      icon = "💰";
    }
    else if (event === "raid") {
      alertType = "raid";
      var viewers = subtitle;
      if (!viewers && data.meta && data.meta.viewers) viewers = data.meta.viewers;
      title = name + " is raiding!";
      detail = viewers + " viewers joining us!";
      icon = "🚀";
    }

    if (!alertType) return;

    showAlert(alertType, title, detail, icon, img, color);
  }

  function showAlert(type, title, detail, icon, img, nameColor) {
    // Create card
    var card = document.createElement("div");
    card.className = "alert-card";
    card.setAttribute("data-type", type);

    // Avatar
    var avatar = document.createElement("img");
    avatar.className = "alert-avatar";
    avatar.src = img;
    avatar.onerror = function() { this.style.display = 'none'; };
    
    // Content
    var content = document.createElement("div");
    content.className = "alert-content";

    var header = document.createElement("div");
    header.className = "alert-header";
    header.textContent = title;
    header.style.color = nameColor;

    var detailDiv = document.createElement("div");
    detailDiv.className = "alert-detail";
    detailDiv.textContent = detail;

    content.appendChild(header);
    content.appendChild(detailDiv);

    // Icon
    var iconDiv = document.createElement("div");
    iconDiv.className = "alert-icon";
    iconDiv.textContent = icon;

    // Progress bar
    var progress = document.createElement("div");
    progress.className = "alert-progress";

    // Assemble
    card.appendChild(iconDiv);
    card.appendChild(content);
    if (img) card.appendChild(avatar);
    card.appendChild(progress);

    // Add to container
    container.appendChild(card);

    // Limit visible alerts
    while (container.children.length > MAX_ALERTS) {
      var old = container.firstChild;
      old.classList.add("removing");
      setTimeout(function() {
        if (old.parentNode) old.parentNode.removeChild(old);
      }, 400);
    }

    // Auto dismiss
    setTimeout(function() {
      card.classList.add("removing");
      setTimeout(function() {
        if (card.parentNode) card.parentNode.removeChild(card);
      }, 400);
    }, 4000);
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
