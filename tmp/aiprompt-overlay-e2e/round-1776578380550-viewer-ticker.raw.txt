<script>
(function () {
  // 1. Setup Bridge
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. State Management
  var currentCounts = {};
  var bar = document.getElementById("viewer-bar");
  var platformsContainer = document.getElementById("platforms-container");
  var totalCountEl = document.getElementById("total-count");

  // 3. Render Logic
  function renderCounts() {
    var keys = Object.keys(currentCounts);
    
    // Clear existing
    platformsContainer.innerHTML = "";

    if (keys.length === 0) {
      bar.classList.remove("visible");
      return;
    }

    bar.classList.add("visible");

    var total = 0;

    keys.forEach(function (platform, index) {
      var count = currentCounts[platform];
      
      // Create Platform Group
      var group = document.createElement("div");
      group.className = "platform-group";

      var nameSpan = document.createElement("span");
      nameSpan.className = "platform-name";
      nameSpan.textContent = platform;

      var countSpan = document.createElement("span");
      countSpan.className = "platform-count";
      countSpan.textContent = count.toLocaleString();

      group.appendChild(nameSpan);
      group.appendChild(countSpan);
      platformsContainer.appendChild(group);

      total += count;
    });

    // Update Total
    totalCountEl.textContent = total.toLocaleString();
  }

  // 4. Message Handler
  function handlePayload(data) {
    // We only care about viewer_updates events
    if (data && data.event === "viewer_updates" && data.meta) {
      // Rule 14: Reset/replace state on snapshot payloads
      currentCounts = {};
      
      // data.meta is an object like { youtube: 800, twitch: 200 }
      for (var platform in data.meta) {
        if (data.meta.hasOwnProperty(platform)) {
          var val = parseInt(data.meta[platform], 10);
          if (!isNaN(val)) {
            currentCounts[platform] = val;
          }
        }
      }
      
      renderCounts();
    }
  }

  // 5. Event Listeners
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
  
  // Initial render (hides bar if no data)
  renderCounts();

})();
</script>
