<script>
(function () {
  // 1. Setup Bridge
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge") || document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // 2. Message Handler
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handleAlert(payload);
    }
  });

  // 3. Logic
  function handleAlert(data) {
    if (!data) return;

    var type = determineAlertType(data);
    if (!type) return;

    var userName = data.chatname || "Unknown User";
    var title = "";
    var subtitle = "";
    var icon = "";

    // Determine content based on type
    switch (type) {
      case "follow":
        title = "New Follower";
        subtitle = "Followed the stream";
        icon = "★";
        break;
      case "sub":
        title = "New Subscriber";
        subtitle = "Subscribed";
        if (data.membership) subtitle += " (" + data.membership + ")";
        icon = "♥";
        break;
      case "gift":
        title = "Gifted Subs";
        subtitle = "Gifted " + (data.subtitle || data.meta && data.meta.gift_count || 1) + " subs";
        icon = "🎁";
        break;
      case "donation":
        title = "Donation";
        subtitle = data.chatmessage || "Generous tip";
        icon = "💰";
        break;
      case "raid":
        title = "Raid";
        subtitle = (data.meta && data.meta.viewer_count) ? "Joining with " + data.meta.viewer_count + " viewers" : "Joined the stream";
        icon = "⚔️";
        break;
    }

    createAlertElement(type, userName, title, subtitle, icon);
  }

  function determineAlertType(data) {
    // Prioritize specific events
    if (data.event === "new_follower") return "follow";
    if (data.event === "raid") return "raid";
    if (data.event === "subscription_gift" || data.event === "gifted") return "gift";
    
    // Donations
    if (data.hasDonation) return "donation";

    // Subscribers (event or membership string)
    if (data.event === "subscriber" || data.membership) return "sub";

    return null;
  }

  function createAlertElement(type, userName, title, subtitle, icon) {
    var container = document.getElementById("alert-container");
    var card = document.createElement("div");
    card.className = "alert-card type-" + type;
    
    // Escape HTML for safety
    var safeUser = escapeHtml(userName);
    var safeTitle = escapeHtml(title);
    var safeSubtitle = escapeHtml(subtitle);

    card.innerHTML = 
      '<div class="alert-icon">' + icon + '</div>' +
      '<div class="alert-content">' +
        '<div class="alert-title">' + safeTitle + '</div>' +
        '<div class="alert-subtitle"><span class="alert-user">' + safeUser + '</span> ' + safeSubtitle + '</div>' +
      '</div>' +
      '<div class="progress-bar"></div>';

    container.appendChild(card);

    // Auto-dismiss after 4 seconds
    setTimeout(function() {
      card.classList.add("dismissing");
      // Remove from DOM after animation finishes
      setTimeout(function() {
        if (card.parentNode) {
          card.parentNode.removeChild(card);
        }
      }, 400); // Matches CSS animation duration
    }, 4000);
  }

  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // Expose for testing
  window.handleOverlayPayload = handleAlert;
})();
</script>
