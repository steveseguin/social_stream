<script>
(function () {
  // Configuration
  var MAX_ROWS = 10;
  var container = document.getElementById("chat-container");
  var rows = []; // Keep track of DOM elements

  // Parse URL params
  var params = new URLSearchParams(location.search);
  if (params.has("limit")) {
    MAX_ROWS = parseInt(params.get("limit"), 10) || 10;
  }

  // Bridge Setup
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_chat_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // Core Handler
  function handlePayload(data) {
    if (!data) return;

    // 1. Filter: Must be chat, not event, not bot
    if (data.event) return; // Skip system events
    if (data.bot === true) return; // Skip bots

    // 2. Extract fields
    var name = data.chatname || "Anonymous";
    var msg = data.chatmessage || "";
    var img = data.chatimg || "";
    var color = data.nameColor || "#4fc3f7";

    // 3. Create DOM
    var row = document.createElement("div");
    row.className = "chat-row";

    // Avatar
    var imgEl = document.createElement("img");
    imgEl.className = "avatar";
    imgEl.src = img;
    imgEl.onerror = function() { this.style.display = 'none'; };
    if (!img) imgEl.style.display = 'none';

    // Content Wrapper
    var contentDiv = document.createElement("div");
    contentDiv.className = "content";

    // Username
    var nameSpan = document.createElement("span");
    nameSpan.className = "username";
    nameSpan.textContent = name;
    nameSpan.style.color = color;

    // Message
    var msgSpan = document.createElement("span");
    msgSpan.className = "message";
    msgSpan.textContent = msg;

    // Assemble
    contentDiv.appendChild(nameSpan);
    contentDiv.appendChild(msgSpan);
    row.appendChild(imgEl);
    row.appendChild(contentDiv);

    // 4. Add to DOM
    container.appendChild(row);
    rows.push(row);

    // 5. Enforce Limit
    while (rows.length > MAX_ROWS) {
      var oldRow = rows.shift();
      if (oldRow && oldRow.parentNode) {
        oldRow.parentNode.removeChild(oldRow);
      }
    }
  }

  // Listen for messages
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
