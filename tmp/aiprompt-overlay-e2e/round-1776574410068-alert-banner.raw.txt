<script>
(function () {
  // --- VDO.Ninja Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_alert_" + Date.now();
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // --- Alert Logic ---

  var container = document.getElementById("alert-container");
  var ALERT_DURATION = 4000; // 4 seconds
  var FADE_DURATION = 500;

  function handlePayload(data) {
    if (!data) return;
    
    // Determine alert type
    var type = getAlertType(data);
    if (!type) return;

    // Extract details
    var user = data.chatname || "Anonymous";
    var message = data.chatmessage || "";
    var amount = data.hasDonation || "";
    var subInfo = data.membership || "";
    var subtitle = data.subtitle || "";
    var img = data.chatimg || "";

    createAlert(type, user, message, amount, subInfo, subtitle, img);
  }

  function getAlertType(data) {
    if (data.event === "new_follower") return "follow";
    if (data.event === "subscription_gift" || data.event === "gifted") return "gift";
    if (data.event === "raid") return "raid";
    if (data.hasDonation) return "donate";
    if (data.event === "subscriber" || data.membership) return "sub";
    return null;
  }

  function createAlert(type, user, message, amount, subInfo, subtitle, img) {
    var card = document.createElement("div");
    card.className = "alert-card alert-" + type;
    
    var icon = getIcon(type);
    var title = getTitle(type);
    
    // Build message content
    var contentHtml = "";
    
    if (type === "donate") {
      contentHtml = formatDonate(user, amount, message);
    } else if (type === "sub") {
      contentHtml = formatSub(user, subInfo, subtitle);
    } else if (type === "gift") {
      contentHtml = formatGift(user, subtitle, message);
    } else if (type === "raid") {
      contentHtml = formatRaid(user, subtitle);
    } else {
      // Follow
      contentHtml = formatFollow(user);
    }

    card.innerHTML = 
      '<div class="alert-icon">' + icon + '</div>' +
      '<div class="alert-content">' +
        '<div class="alert-title">' + escapeHtml(title) + '</div>' +
        '<div class="alert-message">' + contentHtml + '</div>' +
      '</div>';

    container.appendChild(card);

    // Auto-dismiss
    setTimeout(function() {
      if (card.parentNode) {
        card.style.animation = "fadeOut 0.5s ease-in forwards";
        setTimeout(function() {
          if (card.parentNode) {
            card.parentNode.removeChild(card);
          }
        }, FADE_DURATION);
      }
    }, ALERT_DURATION);
  }

  // --- Formatters ---

  function formatFollow(user) {
    return 'New Follower: <span class="highlight">' + escapeHtml(user) + '</span>';
  }

  function formatSub(user, subInfo, subtitle) {
    var text = 'New Subscriber: <span class="highlight">' + escapeHtml(user) + '</span>';
    if (subInfo) text += ' (' + escapeHtml(subInfo) + ')';
    if (subtitle) text += ' - ' + escapeHtml(subtitle);
    return text;
  }

  function formatGift(user, subtitle, message) {
    var text = 'Gifted Sub: <span class="highlight">' + escapeHtml(user) + '</span>';
    if (subtitle) text += ' x' + escapeHtml(subtitle);
    if (message) text += ' - ' + escapeHtml(message);
    return text;
  }

  function formatDonate(user, amount, message) {
    var text = 'Donation: <span class="highlight">' + escapeHtml(user) + '</span> donated <span class="highlight">' + escapeHtml(amount) + '</span>';
    if (message) text += ' — ' + escapeHtml(message);
    return text;
  }

  function formatRaid(user, subtitle) {
    var text = 'Raid: <span class="highlight">' + escapeHtml(user) + '</span> is raiding!';
    if (subtitle) text += ' (' + escapeHtml(subtitle) + ' viewers)';
    return text;
  }

  // --- Icons ---

  function getIcon(type) {
    switch(type) {
      case 'follow': return '👤';
      case 'sub': return '⭐';
      case 'gift': return '🎁';
      case 'donate': return '💰';
      case 'raid': return '⚔️';
      default: return '🔔';
    }
  }

  function getTitle(type) {
    switch(type) {
      case 'follow': return 'New Follower';
      case 'sub': return 'New Subscriber';
      case 'gift': return 'Gifted Subscription';
      case 'donate': return 'Donation Received';
      case 'raid': return 'Incoming Raid';
      default: return 'Alert';
    }
  }

  // --- Utilities ---

  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
