<script>
(function () {
  "use strict";

  /* ---- VDO.Ninja Bridge ---- */
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "viewer_bar_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  /* ---- State ---- */
  var platformCounts = {};   // { youtube: 800, twitch: 200, ... }
  var prevTotal = 0;

  /* ---- DOM refs ---- */
  var bar = document.getElementById("viewer-bar");
  var platformsEl = document.getElementById("platforms");
  var totalCountEl = document.getElementById("total-count");

  /* ---- Helpers ---- */
  function escapeText(str) {
    var div = document.createElement("div");
    div.textContent = str;
    return div.innerHTML;
  }

  function formatNumber(n) {
    return Number(n || 0).toLocaleString("en-US");
  }

  function getTotal() {
    var total = 0;
    var keys = Object.keys(platformCounts);
    for (var i = 0; i < keys.length; i++) {
      total += Number(platformCounts[keys[i]]) || 0;
    }
    return total;
  }

  /* ---- Render ---- */
  function render() {
    var total = getTotal();

    // Build platform list
    var keys = Object.keys(platformCounts).sort();
    platformsEl.innerHTML = "";

    for (var i = 0; i < keys.length; i++) {
      var platform = keys[i];
      var count = Number(platformCounts[platform]) || 0;

      var span = document.createElement("span");
      span.className = "platform-count";

      var labelSpan = document.createElement("span");
      labelSpan.className = "label";
      labelSpan.textContent = platform;

      var countSpan = document.createElement("span");
      countSpan.className = "count";
      countSpan.textContent = formatNumber(count);

      span.appendChild(labelSpan);
      span.appendChild(countSpan);
      platformsEl.appendChild(span);

      // Flash animation if count changed
      if (prevTotal !== total && count > 0) {
        // Trigger reflow then add class
        void countSpan.offsetWidth;
        countSpan.classList.add("flash");
        setTimeout(function (el) {
          el.classList.remove("flash");
        }, 450, countSpan);
      }
    }

    // Update total
    totalCountEl.textContent = formatNumber(total);

    if (prevTotal !== total && total > 0) {
      void totalCountEl.offsetWidth;
      totalCountEl.classList.add("flash");
      setTimeout(function (el) {
        el.classList.remove("flash");
      }, 450, totalCountEl);
    }

    // Show bar on first data
    if (total > 0) {
      bar.classList.remove("hidden");
      bar.classList.add("visible");
    }

    prevTotal = total;
  }

  /* ---- Message Handler ---- */
  function handlePayload(data) {
    if (!data || typeof data !== "object") return;

    // Only handle viewer_updates snapshot payloads
    if (data.event !== "viewer_updates") return;
    if (!data.meta || typeof data.meta !== "object") return;

    // Replace state entirely (snapshot semantics)
    platformCounts = {};
    var keys = Object.keys(data.meta);
    for (var i = 0; i < keys.length; i++) {
      var platform = keys[i];
      var value = data.meta[platform];
      if (typeof value === "number" && value >= 0) {
        platformCounts[platform] = value;
      }
    }

    render();
  }

  // Listen for messages from the bridge iframe
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  // Expose for testing / builder
  window.handleOverlayPayload = handlePayload;

  // Initial render (empty state)
  render();

})();
</script>
