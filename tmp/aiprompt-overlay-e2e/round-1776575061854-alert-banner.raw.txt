<script>
(function() {
  var alertContainer = document.getElementById('alert-container');
  var alertQueue = [];
  var isProcessing = false;

  // VDO.Ninja bridge setup
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function(event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  function handlePayload(data) {
    if (!data || !data.chatname) return;
    
    var alertType = getAlertType(data);
    if (!alertType) return;
    
    var alertData = {
      type: alertType,
      chatname: data.chatname,
      chatmessage: data.chatmessage || "",
      chatimg: data.chatimg || "",
      hasDonation: data.hasDonation || "",
      membership: data.membership || "",
      subtitle: data.subtitle || "",
      event: data.event || "",
      meta: data.meta || {}
    };
    
    queueAlert(alertData);
  }

  function getAlertType(data) {
    if (data.event === "new_follower") return "new_follower";
    if (data.event === "subscriber" || (data.membership && data.event !== "subscription_gift" && data.event !== "gifted")) return "subscriber";
    if (data.event === "subscription_gift" || data.event === "gifted") return "subscription_gift";
    if (data.hasDonation) return "donation";
    if (data.event === "raid") return "raid";
    return null;
  }

  function queueAlert(alertData) {
    alertQueue.push(alertData);
    if (!isProcessing) {
      processQueue();
    }
  }

  function processQueue() {
    if (alertQueue.length === 0) {
      isProcessing = false;
      return;
    }
    
    isProcessing = true;
    var alertData = alertQueue.shift();
    createAlert(alertData);
  }

  function createAlert(alertData) {
    var alertCard = document.createElement('div');
    alertCard.className = 'alert-card';
    alertCard.setAttribute('data-type', alertData.type);
    
    var icon = getAlertIcon(alertData.type);
    var nameClass = 'alert-name';
    var actionClass = 'alert-action';
    var detailText = getAlertDetail(alertData);
    var messageText = alertData.chatmessage ? '<div class="alert-message">' + escapeHtml(alertData.chatmessage) + '</div>' : '';
    
    var avatarHtml = alertData.chatimg ? '<img src="' + escapeHtml(alertData.chatimg) + '" class="alert-avatar" onerror="this.style.display=\'none\'">' : '';
    
    alertCard.innerHTML = 
      avatarHtml +
      '<div class="alert-icon">' + icon + '</div>' +
      '<div class="alert-content">' +
        '<div class="alert-header">' +
          '<span class="' + nameClass + '">' + escapeHtml(alertData.chatname) + '</span>' +
          '<span class="' + actionClass + '">' + getAlertAction(alertData) + '</span>' +
        '</div>' +
        (detailText ? '<div class="alert-detail">' + detailText + '</div>' : '') +
        messageText +
      '</div>' +
      '<div class="progress-bar"></div>';
    
    alertContainer.appendChild(alertCard);
    
    // Remove alert after animation completes
    setTimeout(function() {
      if (alertCard.parentNode) {
        alertCard.parentNode.removeChild(alertCard);
      }
      processQueue();
    }, 4500); // 4s duration + 0.5s exit + 0.5s buffer
  }

  function getAlertIcon(type) {
    switch(type) {
      case 'new_follower': return '👤';
      case 'subscriber': return '⭐';
      case 'subscription_gift': return '🎁';
      case 'donation': return '💰';
      case 'raid': return '🚀';
      default: return '🔔';
    }
  }

  function getAlertAction(alertData) {
    switch(alertData.type) {
      case 'new_follower': return 'followed!';
      case 'subscriber': return 'subscribed!';
      case 'subscription_gift': return 'gifted ' + (alertData.subtitle || alertData.meta.gift_count || 'subs');
      case 'donation': return 'donated ' + escapeHtml(alertData.hasDonation);
      case 'raid': return 'is raiding with ' + (alertData.subtitle || alertData.meta.viewers || 'viewers');
      default: return '';
    }
  }

  function getAlertDetail(alertData) {
    if (alertData.membership) {
      return '<span class="alert-sub">' + escapeHtml(alertData.membership) + '</span>';
    }
    if (alertData.subtitle && alertData.type !== 'subscription_gift' && alertData.type !== 'raid') {
      return '<span class="alert-detail">' + escapeHtml(alertData.subtitle) + '</span>';
    }
    return '';
  }

  function escapeHtml(text) {
    if (!text) return '';
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
  window.sendTestAlert = function() {
    var testAlerts = [
      { chatname: "TestUser1", event: "new_follower", type: "twitch" },
      { chatname: "TestUser2", membership: "Tier 1", type: "youtube" },
      { chatname: "TestUser3", event: "subscription_gift", subtitle: "3", type: "kick" },
      { chatname: "TestUser4", hasDonation: "$25.00", chatmessage: "Great stream!", type: "youtube" },
      { chatname: "TestUser5", event: "raid", subtitle: "50", type: "twitch" }
    ];
    
    testAlerts.forEach(function(alert, index) {
      setTimeout(function() {
        queueAlert(alert);
      }, index * 100);
    });
  };
})();
</script>
