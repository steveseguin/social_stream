<script>
(function () {
  /* ---------- DOM refs ---------- */
  var container = document.getElementById("bar-container");
  var platformStats = document.getElementById("platform-stats");
  var totalCountEl = document.getElementById("total-count");

  /* ---------- State ---------- */
  var currentMeta = {};
  var platformOrder = ["youtube", "twitch", "kick", "tiktok", "facebook", "rumble", "discord"];

  /* ---------- Helpers ---------- */
  function escapeText(str) {
    var div = document.createElement("div");
    div.textContent = str;
    return div.innerHTML;
  }

  function getPlatformColorClass(platform) {
    var cls = "dot-" + platform.toLowerCase();
    var validPlatforms = ["youtube", "twitch", "kick", "tiktok", "facebook", "rumble", "discord"];
    if (validPlatforms.indexOf(platform.toLowerCase()) === -1) {
      cls = "dot-default";
    }
    return cls;
  }

  function formatNumber(n) {
    if (typeof n !== "number" || isNaN(n)) return "0";
    return n.toLocaleString("en-US");
  }

  /* ---------- Render ---------- */
  function render() {
    platformStats.innerHTML = "";
    var total = 0;
    var hasData = false;

    // Render known platforms first
    for (var i = 0; i < platformOrder.length; i++) {
      var key = platformOrder[i];
      if (currentMeta.hasOwnProperty(key)) {
        var val = currentMeta[key];
        if (typeof val === "number" && val > 0) {
          hasData = true;
          total += val;

          var stat = document.createElement("div");
          stat.className = "platform-stat";

          var dot = document.createElement("span");
          dot.className = "platform-dot " + getPlatformColorClass(key);
          stat.appendChild(dot);

          var name = document.createElement("span");
          name.className = "platform-name";
          name.textContent = key;
          stat.appendChild(name);

          var count = document.createElement("span");
          count.className = "platform-count";
          count.textContent = formatNumber(val);
          stat.appendChild(count);

          platformStats.appendChild(stat);
        }
      }
    }

    // Render any unknown platforms
    var knownKeys = {};
    for (var j = 0; j < platformOrder.length; j++) {
      knownKeys[platformOrder[j]] = true;
    }
    for (var key in currentMeta) {
      if (currentMeta.hasOwnProperty(key) && !knownKeys[key]) {
        var val = currentMeta[key];
        if (typeof val === "number" && val > 0) {
          hasData = true;
          total += val;

          var stat = document.createElement("div");
          stat.className = "platform-stat";

          var dot = document.createElement("span");
          dot.className = "platform-dot dot-default";
          stat.appendChild(dot);

          var name = document.createElement("span");
          name.className = "platform-name";
          name.textContent = key;
          stat.appendChild(name);

          var count = document.createElement("span");
          count.className = "platform-count";
          count.textContent = formatNumber(val);
          stat.appendChild(count);

          platformStats.appendChild(stat);
        }
      }
    }

    totalCountEl.textContent = formatNumber(total);

    if (hasData) {
      container.classList.add("visible");
    } else {
      container.classList.remove("visible");
    }
  }

  /* ---------- VDO.Ninja Bridge ---------- */
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_viewerbar_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handlePayload(payload);
    }
  });

  function handlePayload(data) {
    if (!data) return;

    // Only process viewer_updates snapshots
    if (data.event === "viewer_updates" && data.meta) {
      var meta = data.meta;
      if (typeof meta === "object" && !Array.isArray(meta)) {
        currentMeta = {};
        for (var key in meta) {
          if (meta.hasOwnProperty(key)) {
            var val = parseInt(meta[key], 10);
            if (!isNaN(val)) {
              currentMeta[key] = val;
            }
          }
        }
        render();
      }
    }
  }

  window.handleOverlayPayload = handlePayload;

  // Initial render (hidden until data arrives)
  render();
})();
</script>
</body>
