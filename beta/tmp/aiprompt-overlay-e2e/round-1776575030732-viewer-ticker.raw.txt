<script>
(function () {
  // --- State ---
  var platformCounts = {};
  var totalViewers = 0;
  var barVisible = false;

  // --- DOM refs ---
  var barContainer = null;
  var bridge = null;

  // --- Build the bar UI ---
  function buildBar() {
    barContainer = document.createElement('div');
    barContainer.id = 'bar-container';
    barContainer.setAttribute('aria-live', 'polite');
    barContainer.setAttribute('aria-label', 'Live viewer counts');
    document.body.appendChild(barContainer);

    bridge = document.getElementById('ssn_bridge');
  }

  // --- Render the bar ---
  function renderBar() {
    if (!barContainer) return;

    // Clear previous content
    barContainer.innerHTML = '';

    var platforms = Object.keys(platformCounts).sort();
    var total = 0;

    // Build platform count elements
    platforms.forEach(function (platform, index) {
      var count = platformCounts[platform] || 0;
      total += count;

      var platformDiv = document.createElement('div');
      platformDiv.className = 'platform-count';

      var labelSpan = document.createElement('span');
      labelSpan.className = 'label';
      labelSpan.textContent = platform;

      var countSpan = document.createElement('span');
      countSpan.className = 'count';
      countSpan.textContent = count.toLocaleString();

      platformDiv.appendChild(labelSpan);
      platformDiv.appendChild(countSpan);
      barContainer.appendChild(platformDiv);

      // Add divider between platforms and before total
      if (index < platforms.length - 1) {
        var divider = document.createElement('div');
        divider.className = 'divider';
        barContainer.appendChild(divider);
      }
    });

    // Add total if there are any platforms
    if (platforms.length > 0) {
      var totalDivider = document.createElement('div');
      totalDivider.className = 'divider';
      barContainer.appendChild(totalDivider);

      var totalDiv = document.createElement('div');
      totalDiv.className = 'total-count';

      var totalLabel = document.createElement('span');
      totalLabel.className = 'label';
      totalLabel.textContent = 'Total';

      var totalCountSpan = document.createElement('span');
      totalCountSpan.className = 'count';
      totalCountSpan.textContent = total.toLocaleString();

      totalDiv.appendChild(totalLabel);
      totalDiv.appendChild(totalCountSpan);
      barContainer.appendChild(totalDiv);

      // Show the bar
      if (!barVisible) {
        barContainer.classList.add('visible');
        barVisible = true;
      }
    } else {
      // Hide the bar if no data
      if (barVisible) {
        barContainer.classList.remove('visible');
        barVisible = false;
      }
    }
  }

  // --- Handle incoming payloads ---
  function handlePayload(data) {
    if (!data || !data.event || data.event !== 'viewer_updates') return;
    if (!data.meta || typeof data.meta !== 'object') return;

    // Replace state with the snapshot from meta
    platformCounts = {};
    var keys = Object.keys(data.meta);
    for (var i = 0; i < keys.length; i++) {
      var platform = keys[i];
      var count = data.meta[platform];
      if (typeof count === 'number' && count >= 0) {
        platformCounts[platform] = count;
      }
    }

    renderBar();
  }

  // --- Initialize ---
  function init() {
    buildBar();

    // Listen for messages from the bridge
    window.addEventListener('message', function (event) {
      var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
      if (typeof payload !== 'undefined') {
        handlePayload(payload);
      }
    });

    // Expose for testing
    window.handleOverlayPayload = handlePayload;
  }

  // Start when DOM is ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
</script>
