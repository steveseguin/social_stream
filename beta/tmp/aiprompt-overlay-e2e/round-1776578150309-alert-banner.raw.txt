<script>
(function () {
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_banner_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  function handlePayload(data) {
    if (!data || !data.chatname) return;

    var alertType = null;
    var isDonation = false;
    var donationAmount = "";
    var message = "";
    var detail = "";

    // Determine alert type
    if (data.event === "new_follower") {
      alertType = "follower";
      detail = "followed";
    } else if (data.event === "subscriber" || (data.membership && data.event !== "subscription_gift" && data.event !== "gifted")) {
      alertType = "subscriber";
      detail = "subscribed";
      if (data.membership) {
        detail += " (" + data.membership + ")";
      }
    } else if (data.event === "subscription_gift" || data.event === "gifted") {
      alertType = "gifted_sub";
      detail = "gifted a sub";
      if (data.subtitle) {
        detail += " - " + data.subtitle;
      }
    } else if (data.hasDonation) {
      alertType = "donation";
      isDonation = true;
      donationAmount = data.hasDonation;
      message = data.chatmessage || "";
      detail = "donated";
    } else if (data.event === "raid") {
      alertType = "raid";
      detail = "is raiding with";
      if (data.subtitle) {
        detail += " " + data.subtitle + " viewers";
      }
    }

    if (!alertType) return;

    showAlert(data, alertType, isDonation, donationAmount, message, detail);
  }

  function showAlert(data, alertType, isDonation, donationAmount, message, detail) {
    var container = document.getElementById("alert-container");
    var card = document.createElement("div");
    card.className = "alert-card" + (isDonation ? " is-donation" : "");

    // Avatar
    var avatar = document.createElement("img");
    avatar.className = "alert-avatar";
    avatar.src = data.chatimg || "";
    avatar.onerror = function() {
      this.src = "data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100'%3E%3Ccircle cx='50' cy='50' r='50' fill='%23333'/%3E%3Ctext x='50' y='55' font-family='Arial' font-size='40' fill='%23fff' text-anchor='middle'%3E%3F%3C/text%3E%3C/svg%3E";
    };
    card.appendChild(avatar);

    // Content
    var content = document.createElement("div");
    content.className = "alert-content";

    // Header with name and type
    var header = document.createElement("div");
    header.className = "alert-header";

    var nameSpan = document.createElement("span");
    nameSpan.className = "alert-name";
    nameSpan.textContent = data.chatname;

    var typeSpan = document.createElement("span");
    typeSpan.className = "alert-type";
    typeSpan.textContent = alertType.replace("_", " ");

    header.appendChild(nameSpan);
    header.appendChild(typeSpan);
    content.appendChild(header);

    // Message/Detail
    var msgDiv = document.createElement("div");
    msgDiv.className = "alert-message";

    if (isDonation && donationAmount) {
      var amountSpan = document.createElement("span");
      amountSpan.className = "alert-amount";
      amountSpan.textContent = donationAmount;
      msgDiv.appendChild(amountSpan);
    }

    var detailSpan = document.createElement("span");
    detailSpan.textContent = " " + detail;
    msgDiv.appendChild(detailSpan);

    if (message) {
      var msgBreak = document.createElement("br");
      msgDiv.appendChild(msgBreak);
      var msgText = document.createElement("span");
      msgText.textContent = message;
      msgDiv.appendChild(msgText);
    }

    content.appendChild(msgDiv);
    card.appendChild(content);

    // Progress bar
    var progress = document.createElement("div");
    progress.className = "alert-progress";
    card.appendChild(progress);

    container.appendChild(card);

    // Auto-remove after animation
    setTimeout(function() {
      if (card.parentNode) {
        card.parentNode.removeChild(card);
      }
    }, 4000);
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
