<script>
(function () {
  // --- Configuration ---
  const AUTO_DISMISS_MS = 4000;
  const ANIMATION_DURATION_MS = 500; // Must match CSS transition

  // --- DOM Elements ---
  const banner = document.getElementById('alert-banner');
  const avatar = document.getElementById('alert-avatar');
  const nameEl = document.getElementById('alert-name');
  const eventEl = document.getElementById('alert-event');
  const detailsEl = document.getElementById('alert-details');
  const timerEl = document.getElementById('alert-timer');
  const container = document.getElementById('alert-container');

  let currentTimeout = null;
  let currentAnimationFrame = null;
  let isAnimating = false;

  // --- VDO.Ninja Bridge Setup ---
  (function () {
    var params = new URLSearchParams(location.search);
    var roomID = params.get("session") || "";
    var password = params.get("password") || "false";
    var label = params.get("label") || "custom_overlay_" + Date.now();
    var bridge = document.getElementById("ssn_bridge");
    if (roomID) {
      bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
        encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
        "&label=" + encodeURIComponent(label) +
        "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
    }
    window.addEventListener("message", function (event) {
      var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
      if (typeof payload !== "undefined") handlePayload(payload);
    });
    window.handleOverlayPayload = handlePayload; // For testing
  }());

  // --- Helper: Escape HTML ---
  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
  }

  // --- Helper: Get Initials ---
  function getInitials(name) {
    if (!name) return "?";
    return name.charAt(0).toUpperCase();
  }

  // --- Core Logic: Process Payload ---
  function handlePayload(data) {
    if (!data) return;

    // Filter: Only process specific events
    const eventType = data.event;
    const hasDonation = !!data.hasDonation;
    const hasMembership = !!data.membership;
    const isGifted = (eventType === "subscription_gift" || eventType === "gifted");
    const isSub = (eventType === "subscriber" || (hasMembership && !eventType));
    const isFollow = (eventType === "new_follower");
    const isRaid = (eventType === "raid");

    // Check if this is a valid alert type
    if (!isFollow && !isSub && !isGifted && !hasDonation && !isRaid) {
      return; // Not an alert we handle
    }

    // Prevent spamming if an animation is currently playing
    if (isAnimating) return;

    // Determine Banner Type & Content
    let type = "";
    let eventText = "";
    let detailText = "";
    let avatarUrl = data.chatimg;

    if (isFollow) {
      type = "follow";
      eventText = "New Follower";
      detailText = "Thanks for following!";
    } else if (isGifted) {
      type = "gifted";
      eventText = "Gifted Subs";
      // Try to get count from subtitle or meta
      const count = data.subtitle || (data.meta && data.meta.gift_count) || "multiple";
      detailText = `Gifted ${escapeHtml(String(count))} subscriptions`;
    } else if (isSub) {
      type = "sub";
      eventText = "Subscriber";
      detailText = data.membership || "Subscribed";
    } else if (hasDonation) {
      type = "donation";
      eventText = "Donation";
      detailText = `${escapeHtml(data.hasDonation)} — ${data.chatmessage || "Thank you!"}`;
    } else if (isRaid) {
      type = "raid";
      eventText = "Raid";
      // Try to get viewer count from subtitle or meta
      const viewers = data.subtitle || (data.meta && data.meta.viewers) || "viewers";
      detailText = `Joining with ${escapeHtml(String(viewers))} viewers`;
    }

    // Update DOM
    banner.setAttribute('data-type', type);
    nameEl.textContent = data.chatname || "Unknown";
    eventEl.textContent = eventText;
    detailsEl.textContent = detailText;

    // Handle Avatar
    if (avatarUrl) {
      avatar.style.backgroundImage = `url('${avatarUrl}')`;
      avatar.textContent = ""; // Clear initials
    } else {
      avatar.style.backgroundImage = "none";
      avatar.textContent = getInitials(data.chatname);
    }

    // Trigger Animation
    triggerAnimation();
  }

  function triggerAnimation() {
    isAnimating = true;
    
    // Reset classes
    banner.classList.remove('exit');
    banner.classList.add('active');

    // Start Timer Animation
    timerEl.style.transition = 'none';
    timerEl.style.transform = 'scaleX(1)';
    
    // Force reflow
    void timerEl.offsetWidth;

    // Animate bar shrinking
    timerEl.style.transition = `transform ${AUTO_DISMISS_MS}ms linear`;
    timerEl.style.transform = 'scaleX(0)';

    // Set Auto-Dismiss
    if (currentTimeout) clearTimeout(currentTimeout);
    
    currentTimeout = setTimeout(() => {
      dismissAlert();
    }, AUTO_DISMISS_MS);
  }

  function dismissAlert() {
    banner.classList.remove('active');
    banner.classList.add('exit');

    // Wait for exit animation to finish before allowing new alerts
    setTimeout(() => {
      isAnimating = false;
      // Optional: Reset timer bar for next time
      timerEl.style.transition = 'none';
      timerEl.style.transform = 'scaleX(1)';
    }, ANIMATION_DURATION_MS);
  }

})();
</script>
</body>
