<script>
(function () {
  // --- Configuration ---
  const AUTO_DISMISS_TIME = 4000; // 4 seconds
  
  // --- State ---
  const container = document.getElementById('alert-container');
  let isProcessing = false; // Queue management
  const alertQueue = [];

  // --- VDO.Ninja Bridge Setup ---
  (function initBridge() {
    var params = new URLSearchParams(location.search);
    var roomID = params.get("session") || "";
    var password = params.get("password") || "false";
    var label = params.get("label") || "custom_overlay_alert_" + Date.now();
    var bridge = document.getElementById("ssn_bridge");
    
    if (roomID) {
      bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
        encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
        "&label=" + encodeURIComponent(label) +
        "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
    }

    window.addEventListener("message", function (event) {
      var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
      if (typeof payload !== "undefined") {
        handlePayload(payload);
      }
    });

    window.handleOverlayPayload = handlePayload;
  })();

  // --- Core Logic ---

  function handlePayload(data) {
    if (!data) return;

    // 1. Filter: Only process specific alert types
    const isFollower = data.event === "new_follower";
    const isSubscriber = data.event === "subscriber" || (data.membership && data.event !== "subscription_gift");
    const isGifted = data.event === "subscription_gift" || data.event === "gifted";
    const isDonation = !!data.hasDonation;
    const isRaid = data.event === "raid";

    if (!isFollower && !isSubscriber && !isGifted && !isDonation && !isRaid) {
      return; // Not an alert we care about
    }

    // 2. Determine Type & Content
    let type = "";
    let icon = "";
    let title = "";
    let detail = "";
    let user = data.chatname || "Anonymous";

    if (isFollower) {
      type = "new_follower";
      icon = "👤";
      title = "New Follower";
      detail = user + " just followed!";
    } else if (isSubscriber) {
      type = "subscriber";
      icon = "⭐";
      title = "New Subscriber";
      detail = user + " subscribed";
      if (data.membership) detail += " (" + data.membership + ")";
    } else if (isGifted) {
      type = "subscription_gift";
      icon = "🎁";
      title = "Gifted Subs";
      let count = data.subtitle || data.meta && data.meta.gift_count || "a sub";
      detail = user + " gifted " + count;
    } else if (isDonation) {
      type = "donation";
      icon = "💰";
      title = "Donation";
      detail = user + " donated " + data.hasDonation;
      if (data.chatmessage) detail += ": " + data.chatmessage;
    } else if (isRaid) {
      type = "raid";
      icon = "🚀";
      title = "Raid Incoming!";
      let viewers = data.subtitle || data.meta && data.meta.viewers || "viewers";
      detail = user + " is raiding with " + viewers;
    }

    // 3. Add to Queue
    alertQueue.push({ type, icon, title, detail, user });
    processQueue();
  }

  function processQueue() {
    if (isProcessing || alertQueue.length === 0) return;
    isProcessing = true;

    const alertData = alertQueue.shift();
    createAlertElement(alertData);
  }

  function createAlertElement(data) {
    const el = document.createElement('div');
    el.className = 'alert-card';
    el.setAttribute('data-type', data.type);
    
    // HTML Content
    el.innerHTML = `
      <div class="alert-icon">${data.icon}</div>
      <div class="alert-content">
        <div class="alert-header">
          <span>${data.title}</span>
          <span class="alert-user">${escapeHtml(data.user)}</span>
        </div>
        <div class="alert-detail">${escapeHtml(data.detail)}</div>
      </div>
      <div class="alert-progress"></div>
    `;

    container.appendChild(el);

    // Auto-dismiss logic
    setTimeout(function() {
      el.classList.add('removing');
      el.addEventListener('animationend', function() {
        if (el.parentNode) {
          el.parentNode.removeChild(el);
        }
        isProcessing = false;
        processQueue(); // Trigger next in queue
      });
    }, AUTO_DISMISS_TIME);
  }

  // --- Utilities ---
  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement('div');
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // --- Test Mode Helper ---
  // If no session ID is provided, you can manually trigger alerts via console:
  // window.testAlert('new_follower', 'TestUser')
  // window.testAlert('donation', 'RichUser', '$50.00')
  window.testAlert = function(type, user, extra) {
    let data = { chatname: user, event: type };
    if (type === 'donation') data.hasDonation = extra || "$10.00";
    if (type === 'subscriber') data.membership = "Tier 1";
    if (type === 'subscription_gift') data.subtitle = "3";
    handlePayload(data);
  };

})();
</script>
