<script>
(function () {
  // --- Configuration ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "custom_overlay_chat_" + Date.now();
  var limit = parseInt(params.get("limit")) || 10;
  
  var container = document.getElementById("chat-container");
  var messages = []; // Store valid messages to manage the limit

  // --- Bridge Setup ---
  var bridge = document.getElementById("ssn_bridge");
  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // --- Core Logic ---
  function handlePayload(data) {
    // 1. Filter: Must be chat (no event) and not a bot
    if (data.event || data.bot) return;
    if (!data.chatname || !data.chatmessage) return;

    // 2. Add to state
    messages.push(data);

    // 3. Enforce limit
    while (messages.length > limit) {
      messages.shift();
    }

    // 4. Render
    renderChat();
  }

  function escapeHtml(unsafe) {
    if (!unsafe) return "";
    return unsafe
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#039;");
  }

  function renderChat() {
    // Clear current DOM
    container.innerHTML = "";

    // Rebuild from state
    messages.forEach(function(msg) {
      var row = document.createElement("div");
      row.className = "chat-row";

      // Avatar
      var img = document.createElement("img");
      img.className = "avatar";
      img.src = msg.chatimg || "";
      img.alt = msg.chatname;
      // Fallback if image fails
      img.onerror = function() { this.src = "data:image/svg+xml;base64,PHN2ZyB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciIHZpZXdCb3g9IjAgMCAyNCAyNCIgZmlsbD0iIzMzMyI+PHBhdGggZD0iTTEyIDEyYzIuMjEgMCA0LTEuNzkgNC00cy0xLjc5LTQtNC00LTQgMS43OS00IDQgMS43OSA0IDQgNHptMCAyYy0yLjY3IDAtOCAxLjM0LTggNHYyaDE2di0yYzAtMi42Ni01LjMzLTQtOC00eiIvPjwvc3ZnPg=="; };

      // Content
      var contentDiv = document.createElement("div");
      contentDiv.className = "message-content";

      var nameSpan = document.createElement("span");
      nameSpan.className = "username";
      nameSpan.textContent = msg.chatname;
      if (msg.nameColor) nameSpan.style.color = msg.nameColor;

      var textSpan = document.createElement("span");
      textSpan.className = "text";
      textSpan.textContent = msg.chatmessage;

      contentDiv.appendChild(nameSpan);
      contentDiv.appendChild(textSpan);

      row.appendChild(img);
      row.appendChild(contentDiv);
      container.appendChild(row);
    });
  }

  // --- Event Listeners ---
  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
