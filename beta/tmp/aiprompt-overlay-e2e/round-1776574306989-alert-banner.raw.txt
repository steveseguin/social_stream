<script>
(function() {
  // --- VDO.Ninja Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_banner_" + Date.now();
  
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // --- Alert Logic ---
  var container = document.getElementById("alert-container");
  var activeAlerts = [];

  function handlePayload(data) {
    if (!data) return;

    var type = data.type || "";
    var event = data.event || "";
    var name = data.chatname || "Unknown";
    var msg = data.chatmessage || "";
    var donation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";

    var alertType = null;
    var title = "";
    var message = "";
    var sub = "";
    var icon = "";

    // 1. New Follower
    if (event === "new_follower") {
      alertType = "new_follower";
      title = "New Follower";
      message = name + " just followed!";
      icon = "★";
    }
    // 2. Subscriber (Resub)
    else if (event === "subscriber" || (membership && !event)) {
      alertType = "subscriber";
      title = "New Subscriber";
      message = name + " subscribed";
      if (membership) message += " (" + membership + ")";
      icon = "♥";
    }
    // 3. Gifted Sub
    else if (event === "subscription_gift" || event === "gifted") {
      alertType = "subscription_gift";
      title = "Gifted Sub";
      message = name + " gifted";
      if (subtitle) message += " " + subtitle;
      else if (data.meta && data.meta.gift_count) message += " " + data.meta.gift_count + " subs";
      icon = "🎁";
    }
    // 4. Donation
    else if (donation) {
      alertType = "donation";
      title = "Donation";
      message = name + " donated " + donation;
      if (msg) sub = '"' + msg + '"';
      icon = "💰";
    }
    // 5. Raid
    else if (event === "raid") {
      alertType = "raid";
      title = "Raid Incoming!";
      message = name + " is raiding with " + (data.subtitle || "viewers");
      icon = "⚔️";
    }

    if (alertType) {
      showAlert(alertType, title, message, sub, icon);
    }
  }

  function showAlert(type, title, message, sub, icon) {
    var el = document.createElement("div");
    el.className = "alert-card";
    el.setAttribute("data-type", type);
    
    var html = 
      '<div class="alert-icon">' + icon + '</div>' +
      '<div class="alert-content">' +
        '<div class="alert-title">' + escapeHtml(title) + '</div>' +
        '<div class="alert-message">' + escapeHtml(message) + '</div>' +
        (sub ? '<div class="alert-sub">' + escapeHtml(sub) + '</div>' : '') +
      '</div>';
      
    el.innerHTML = html;
    container.appendChild(el);

    // Auto dismiss after 4 seconds
    setTimeout(function() {
      dismissAlert(el);
    }, 4000);
  }

  function dismissAlert(el) {
    if (!el || !el.parentNode) return;
    el.classList.add("dismissing");
    el.addEventListener("animationend", function() {
      if (el.parentNode) el.parentNode.removeChild(el);
    });
  }

  function escapeHtml(text) {
    if (!text) return "";
    var div = document.createElement("div");
    div.appendChild(document.createTextNode(text));
    return div.innerHTML;
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
</body>
