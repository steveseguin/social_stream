<script>
(function () {
  "use strict";

  /* ---- state ---- */
  var viewerState = {};
  var platformOrder = [];
  var lastTotal = 0;

  /* ---- DOM refs ---- */
  var barEl = document.getElementById("viewer-bar");
  var platformsContainer = document.getElementById("platforms-container");
  var totalCountEl = document.getElementById("total-count");

  /* ---- platform color palette ---- */
  var platformColors = [
    "#ff4444", "#9146ff", "#53fc18", "#fe2c55",
    "#0077ed", "#00c74c", "#ffaa00", "#00bfff",
    "#ff69b4", "#a855f7", "#22c55e", "#f97316",
    "#eab308", "#14b8a6", "#ec4899", "#6366f1"
  ];

  function getPlatformColor(name) {
    var idx = 0;
    var seen = {};
    for (var i = 0; i < platformOrder.length; i++) {
      seen[platformOrder[i]] = true;
    }
    var colorIdx = 0;
    for (var k = 0; k < platformOrder.length; k++) {
      if (platformOrder[k] === name) {
        return platformColors[colorIdx % platformColors.length];
      }
      colorIdx++;
    }
    return platformColors[platformColors.length - 1];
  }

  /* ---- build / rebuild platform chips ---- */
  function renderPlatforms() {
    platformsContainer.innerHTML = "";

    for (var i = 0; i < platformOrder.length; i++) {
      var plat = platformOrder[i];
      var count = viewerState[plat] || 0;
      if (count === 0) continue;

      var chip = document.createElement("div");
      chip.className = "platform-count";

      var dot = document.createElement("span");
      dot.className = "dot";
      dot.style.backgroundColor = getPlatformColor(plat);

      var label = document.createElement("span");
      label.className = "label";
      label.textContent = plat;

      var countSpan = document.createElement("span");
      countSpan.className = "count";
      countSpan.id = "count-" + plat;
      countSpan.textContent = count.toLocaleString();

      chip.appendChild(dot);
      chip.appendChild(label);
      chip.appendChild(countSpan);
      platformsContainer.appendChild(chip);
    }
  }

  /* ---- bump animation ---- */
  function bumpCount(el) {
    if (!el) return;
    el.classList.add("bump");
    setTimeout(function () {
      el.classList.remove("bump");
    }, 160);
  }

  /* ---- update total ---- */
  function updateTotal() {
    var total = 0;
    for (var k in viewerState) {
      if (viewerState.hasOwnProperty(k)) {
        total += viewerState[k] || 0;
      }
    }

    totalCountEl.textContent = total.toLocaleString();

    if (total !== lastTotal) {
      bumpCount(totalCountEl);
      lastTotal = total;
    }

    if (total > 0) {
      barEl.classList.add("visible");
    } else {
      barEl.classList.remove("visible");
    }
  }

  /* ---- handle incoming payload ---- */
  function handlePayload(data) {
    if (!data || typeof data !== "object") return;

    if (data.event === "viewer_updates" && data.meta && typeof data.meta === "object") {
      /*
        meta is a FULL SNAPSHOT — replace state entirely.
        Do NOT merge with old data.
      */
      var newOrder = [];
      var newState = {};

      for (var key in data.meta) {
        if (data.meta.hasOwnProperty(key)) {
          var val = parseInt(data.meta[key], 10);
          if (!isNaN(val) && val > 0) {
            newState[key] = val;
            newOrder.push(key);
          }
        }
      }

      viewerState = newState;
      platformOrder = newOrder;

      renderPlatforms();
      updateTotal();
    }
  }

  /* ---- VDO.Ninja bridge ---- */
  (function initBridge() {
    var params = new URLSearchParams(location.search);
    var roomID = params.get("session") || "";
    var password = params.get("password") || "false";
    var label = params.get("label") || "custom_overlay_viewerbar_" + Date.now();
    var bridge = document.getElementById("ssn_bridge");

    if (roomID) {
      bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
        encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
        "&label=" + encodeURIComponent(label) +
        "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
    }

    window.addEventListener("message", function (event) {
      var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
      if (typeof payload !== "undefined") {
        handlePayload(payload);
      }
    });

    window.handleOverlayPayload = handlePayload;
  })();

})();
</script>
