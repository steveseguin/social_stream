<script>
(function () {
  // --- VDO.Ninja Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "alert_banner_" + Date.now();
  var bridge = document.createElement("iframe");
  bridge.id = "ssn_bridge";
  bridge.style.display = "none";
  document.body.appendChild(bridge);

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  window.addEventListener("message", function (event) {
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") handlePayload(payload);
  });

  // --- Alert Logic ---
  var container = document.getElementById("alert-container");
  var alertTimeout = 4000; // 4 seconds

  function handlePayload(data) {
    if (!data) return;

    var type = data.type || "";
    var eventName = data.event || "";
    var chatName = data.chatname || "Anonymous";
    var message = data.chatmessage || "";
    var donation = data.hasDonation || "";
    var membership = data.membership || "";
    var subtitle = data.subtitle || "";

    var alertType = null;
    var title = "";
    var subText = "";
    var icon = "";

    // Determine alert type and content
    if (eventName === "new_follower") {
      alertType = "new_follower";
      title = "New Follower!";
      subText = chatName + " just followed.";
      icon = "★";
    } 
    else if (eventName === "subscriber" || (membership && !eventName)) {
      alertType = "subscriber";
      title = "New Subscriber!";
      subText = membership ? membership : chatName + " subscribed.";
      icon = "♥";
    }
    else if (eventName === "subscription_gift" || eventName === "gifted") {
      alertType = "gifted";
      title = "Gifted Subs!";
      subText = subtitle ? subtitle + " gifted subs" : chatName + " gifted subs";
      icon = "🎁";
    }
    else if (eventName === "raid") {
      alertType = "raid";
      title = "Raider!";
      subText = chatName + " is raiding with " + (subtitle || "viewers");
      icon = "⚔️";
    }
    else if (donation) {
      alertType = "donation";
      title = "Donation!";
      subText = chatName + " donated " + donation;
      if (message) subText += " — \"" + message + "\"";
      icon = "💰";
    }

    if (alertType) {
      createAlert(alertType, title, subText, icon);
    }
  }

  function createAlert(type, title, subText, icon) {
    var card = document.createElement("div");
    card.className = "alert-card";
    card.setAttribute("data-type", type);

    var iconSpan = document.createElement("span");
    iconSpan.className = "icon";
    iconSpan.textContent = icon;

    var detailsDiv = document.createElement("div");
    detailsDiv.className = "details";

    var titleSpan = document.createElement("div");
    titleSpan.textContent = title;

    var subSpan = document.createElement("div");
    subSpan.className = "sub-text";
    subSpan.textContent = subText;

    detailsDiv.appendChild(titleSpan);
    detailsDiv.appendChild(subSpan);
    card.appendChild(iconSpan);
    card.appendChild(detailsDiv);

    container.appendChild(card);

    // Auto-dismiss
    setTimeout(function () {
      card.classList.add("fade-out");
      card.addEventListener("animationend", function () {
        if (card.parentNode) {
          card.parentNode.removeChild(card);
        }
      });
    }, alertTimeout);
  }

  // Expose for testing
  window.handleOverlayPayload = handlePayload;
})();
</script>
</body>
