Here is the polished chat overlay. It features a clean, modern design with a dark semi-transparent background, rounded avatars, distinct typography for usernames and messages, and a smooth slide-in animation.

<script>
(function() {
  // --- Configuration ---
  const MAX_ITEMS = 15; // Max messages to keep in DOM
  const FAKE_AVATAR = "data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100'%3E%3Crect width='100' height='100' fill='%23555'/%3E%3Ctext x='50' y='50' dy='.35em' fill='white' text-anchor='middle' font-size='40'%3E?%3C/text%3E%3C/svg%3E";

  // --- Bridge Setup ---
  var params = new URLSearchParams(location.search);
  var roomID = params.get("session") || "";
  var password = params.get("password") || "false";
  var label = params.get("label") || "chat_overlay_" + Date.now();
  var bridge = document.getElementById("ssn_bridge");
  
  // Create bridge if not exists
  if (!bridge) {
    bridge = document.createElement("iframe");
    bridge.id = "ssn_bridge";
    bridge.style.display = "none";
    document.body.appendChild(bridge);
  }

  if (roomID) {
    bridge.src = "https://vdo.socialstream.ninja/?ln&salt=vdo.ninja&password=" +
      encodeURIComponent(password) + "&view=" + encodeURIComponent(roomID) +
      "&label=" + encodeURIComponent(label) +
      "&noaudio&novideo&cleanoutput&room=" + encodeURIComponent(roomID);
  }

  // --- Core Logic ---
  window.handleOverlayPayload = function(data) {
    if (!data || !data.chatname) return;

    // Filter: Only show chat messages (ignore system events unless they have chat content)
    // We allow events if they have a chatname and message
    if (!data.chatmessage && !data.event) return;

    // Filter: Hide bots if requested via URL
    if (params.get("hidetype") === "bots" && data.bot) return;

    // Filter: Only donations if requested
    if (params.get("donationsonly") && !data.hasDonation) return;

    renderChat(data);
  };

  function renderChat(data) {
    const feed = document.getElementById("feed");
    
    // Create Element
    const item = document.createElement("div");
    item.className = "chat-item";
    if (data.hasDonation) item.classList.add("has-donation");

    // Avatar
    const img = document.createElement("img");
    img.className = "avatar";
    img.src = data.chatimg || FAKE_AVATAR;
    img.onerror = function() { this.src = FAKE_AVATAR; };
    item.appendChild(img);

    // Content Wrapper
    const content = document.createElement("div");
    content.className = "content";

    // Header (Name + Badges)
    const header = document.createElement("div");
    header.className = "header";

    const nameSpan = document.createElement("span");
    nameSpan.className = "username";
    nameSpan.textContent = data.chatname;
    if (data.nameColor) nameSpan.style.color = data.nameColor;
    header.appendChild(nameSpan);

    // Badges (Simple representation)
    if (data.chatbadges && Array.isArray(data.chatbadges)) {
      const badgeContainer = document.createElement("span");
      badgeContainer.className = "badges";
      data.chatbadges.forEach(url => {
        if (url) {
          const b = document.createElement("img");
          b.src = url;
          b.className = "badge";
          b.title = "Badge";
          badgeContainer.appendChild(b);
        }
      });
      header.appendChild(badgeContainer);
    }

    content.appendChild(header);

    // Message Body
    const msgDiv = document.createElement("div");
    msgDiv.className = "message";
    
    // Handle Donation Tag
    if (data.hasDonation) {
      const tag = document.createElement("span");
      tag.className = "donation-tag";
      tag.textContent = data.hasDonation;
      msgDiv.appendChild(tag);
    }

    // Handle Message Text
    const textNode = document.createTextNode(data.chatmessage);
    msgDiv.appendChild(textNode);
    
    content.appendChild(msgDiv);
    item.appendChild(content);

    // Prepend to feed (newest top)
    feed.insertBefore(item, feed.firstChild);

    // Cleanup old items
    while (feed.children.length > MAX_ITEMS) {
      const last = feed.lastChild;
      // Animate out last item before removing
      last.style.animation = "fadeOut 0.5s forwards";
      setTimeout(() => { if(last.parentNode) last.parentNode.removeChild(last); }, 500);
    }
  }

  // --- Event Listener ---
  window.addEventListener("message", function(event) {
    // Handle data from VDO.Ninja bridge
    var payload = event.data && event.data.dataReceived && event.data.dataReceived.overlayNinja;
    if (typeof payload !== "undefined") {
      handleOverlayPayload(payload);
    }
  });

})();
</script>
</body>
